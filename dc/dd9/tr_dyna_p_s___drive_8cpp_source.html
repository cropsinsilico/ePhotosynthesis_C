63void trDynaPSDriver::setup() {
64
65 if (ParaNum <= 103) {
66 inputVars->PSRatio[ParaNum] = Ratio;
67 } else if (ParaNum <= 169) {
68 inputVars->SUCSRatio[ParaNum - 103] = Ratio;
69 } else if (ParaNum <= 217) {
70 inputVars->PRRatio[ParaNum - 169] = Ratio;
71 } else if (ParaNum <= 233) {
72 inputVars->RuACTRatio[ParaNum - 217] = Ratio;
73 } else if (ParaNum <= 256) {
74 inputVars->FIRatio[ParaNum - 233] = Ratio;
75 } else if (ParaNum <= 305) {
76 inputVars->BFRatio[ParaNum - 256] = Ratio;
77 } else if (ParaNum <= 309) {
78 inputVars->XanCycleRatio[ParaNum - 305] = Ratio;
79 }
80 SYSInitial(inputVars);
81
82 IniModelCom(inputVars); // Initialize the structure of the model, i.e. Is this model separate or combined with others.
83
84 // The combination of BF and FI model
85 inputVars->BF_FI_com = true;
86
87 // true means that the overall EPS model is used. false means partial model of FIBF is used.
88 inputVars->FIBF_PSPR_com = true;
89
90 // A global variable to indicate whether the RuACT is run by itself or combined with others.
91 inputVars->RuACT_EPS_com = true; // Since this is run within this program, it is combinbed, therefore, it is assigned value true, otherwise, assign value false.
92
93 // This is the connection between Redox and RA.
94 // langmm: I believe this should be true, but the RedoxReg module
95 // solver problem does not appear to be well solved and is always
96 // solved by the initial guess.
97 inputVars->RedoxReg_RA_com = false; // This means that the connection is there.
98
99 inputVars->XanCycle_BF_com = true;
100
101 inputVars->RROEA_EPS_com = true;
102
103 inputVars->EPS_SUCS_com = true;
104
105 // This is a variable indicating whether the PSPR model is actually need to be combined with SUCS or not. If 1 then means combined; 0 means not.
106 inputVars->PSPR_SUCS_com = true;
107
108 // Next is to initialize the vector.
109 trDynaPSCondition* trDynaPS_con = trDynaPS_Ini();
110 ParamSet(inputVars);
111 constraints = trDynaPS_con->toArray();
112 delete trDynaPS_con;
113
114}

116void trDynaPSDriver::getResults() {
117 // call the functions one last time to get the correct values we need
118 trDynaPSCondition* trDynaPS_res = new trDynaPSCondition(intermediateRes);
119 arr temp = trDynaPS::MB(time, trDynaPS_res, inputVars);
120
121 double CarbonRate = inputVars->RuACT_Vel.v6_1 * inputVars->AVR;
122 double VPR = inputVars->RuACT_Vel.v6_2 * inputVars->AVR;
123 double Vpgasink = inputVars->SUCS_Vel.vpga_use * inputVars->AVR;
124 double VStarch = (inputVars->PS_Vel.v23 - inputVars->PS_Vel.v25) * inputVars->AVR;
125 double Vt3p = (inputVars->PS_Vel.v31 + inputVars->PS_Vel.v33) * inputVars->AVR;
126 const double CO2AR = TargetFunVal(inputVars);
127 results = zeros(8);
128 results[0] = CarbonRate; //Vc
129 results[1] = VPR; //Vo
130 results[2] = Vpgasink; //PGA
131 results[3] = Vt3p; //VT3P
132 results[4] = VStarch; //Vstarch
133 results[5] = inputVars->PR_Vel.v1in * inputVars->AVR; //Vt_glycerate
134 results[6] = inputVars->PR_Vel.v2out * inputVars->AVR; //Vt_glycolate
135 results[7] = CO2AR;
136
137 if(inputVars->record) {
138 makeFluxTR(inputVars);
139 }
140 delete trDynaPS_res;
141 IniModelCom(inputVars);
142}
