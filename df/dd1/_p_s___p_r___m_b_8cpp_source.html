37PS_PRCondition* PS_PR::_MB_con(const double t, const PS_PRCondition* const PS_PR_con, Variables *theVars) {
38
39 const double vATPcost = theVars->TestATPCost / theVars->AVR;
40
41 PRCondition* PR_con = PS_PR_con->PR_con;
42 PSCondition* PS_con = PS_PR_con->PS_con;
43 PR::setPGA(PS_con->PGA);
44 const double PR2PS_Pgca = PS_PR_con->PR_con->PGCA; // FOr transfering information between PR to PS.
45
47 // 2. Add exprimental conditions here; Conditions like light, temperature, CO2, O2 concentration should be added here //
49
50 Condition(t, theVars);
51
52 arr PS_Param = zeros(2);
53 PS_Param[0] = theVars->PS_PR_Param;
54 PS_Param[1] = PR2PS_Pgca;
55
56 PS::setParam(PS_Param);
57 PSCondition* PSdydt = PS::MB_con(t, PS_con, theVars);
58
59 arr PR_Param = zeros(2);
60 PR_Param[0] = theVars->PS_PR_Param; // To indicate that the calcualtion is using the combined model
61 // for the PS-PR combined model. 0: Combined model; 1: Separate model
62 PR_Param[1] = theVars->Pi;
63
64 PRCondition* PRdydt = PR::MB_con(t, PR_con, theVars);
65
66 // Assign the rate of reaction that is calculated from the photosynthesis and photorespiration routine.
68 // 5. Calculation of the mass balance equations ////
70
71 PSdydt->RuBP = theVars->PS_Vel.v13 - theVars->PS_Vel.v1 - theVars->PR_Vel.v111;
72 PSdydt->PGA = 2 * theVars->PS_Vel.v1 - theVars->PS_Vel.v2 - theVars->PS_Vel.v32 +
73 theVars->PR_Vel.v113 + theVars->PR_Vel.v111;
74
75 if (theVars->useC3) {
76 PSdydt->ATP = theVars->PS_Vel.v16 - theVars->PS_Vel.v2 - theVars->PS_Vel.v23 -
77 theVars->PS_Vel.v13 - theVars->PR_Vel.v113;
78 } else {
79 PSdydt->ATP = theVars->PS_Vel.v16 - theVars->PS_Vel.v2 - theVars->PS_Vel.v23 -
80 theVars->PS_Vel.v13 - theVars->PR_Vel.v113 - theVars->PS_Vel.v25 -
81 theVars->PR_Vel.v131 - vATPcost;//WY202003 extra ATP cost
82 }
83
84 if (theVars->useC3)
85 PSdydt->HexP = theVars->PS_Vel.v6 - theVars->PS_Vel.v7 - theVars->PS_Vel.v23;
86
87 PS_PRCondition *dydt = new PS_PRCondition(PSdydt, PRdydt);
88#ifdef INCDEBUG
89 DEBUG_DELTA(dydt)
90#endif
91 return dydt;
92}
