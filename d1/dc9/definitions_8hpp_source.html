355// insert a new data set into the array
356// If step already exists then replace the existing entry, otherwise
357// add the new data to the vector
358template <typename T>
359void TimeSeries<T>::insert(std::size_t step, double time, T &input) {
360 std::vector<int>::iterator it = std::find(_step.begin(), _step.end(), step);
361 T vec(input);
362 if (it == _step.end()) {
363 _step.push_back(static_cast<int>(step));
364 _timestamp.push_back(time);
365 _data.push_back(vec);
366 } else {
367 std::size_t index = std::distance(_step.begin(), it);
368 _timestamp[index] = time;
369 _data[index] = vec;
370 }
371}
372
373// write the contents of the class to the stream
374template <typename T>
375void TimeSeries<T>::write(std::ofstream &of) {
376 for (std::size_t i = 0; i < _step.size(); i++) {
377 of << _timestamp[i];
378 for (const auto &e : _data[i])
379 of << "," << e;
380 of << std::endl;
381 }
382}

384template <typename T>
385bool TimeSeries<T>::operator==(const TimeSeries<T>& other) const {
386 if (current != other.current) return false;
387#define CHECK(name) \
388 if (name.size() != other.name.size()) \
389 return false; \
390 for (std::size_t i = 0; i < name.size(); i++) { \
391 if (name[i] != other.name[i]) return false; \
392 }
393 CHECK(_data);
394 CHECK(_step);
395 CHECK(_timestamp);
396#undef CHECK
397 return true;
398}
