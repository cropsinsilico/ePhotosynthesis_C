157 template<typename T1>
158 bool enum_value_search(const std::map<T1, std::string> map,
159 const std::string& val,
160 T1& key, bool allow_anycase=false,
161 std::string prefix="",
162 std::string suffix="") {
163 for (typename std::map<T1, std::string>::const_iterator it = map.cbegin();
164 it != map.cend(); it++) {
165 if ((it->second == val) ||
166 (allow_anycase && (str_toupper(val) == str_toupper(it->second))) ||
167 ((!prefix.empty()) && (val == (prefix + it->second))) ||
168 ((!suffix.empty()) && (val == (it->second + suffix)))) {
169 key = it->first;
170 return true;
171 }
172 }
173 return false;

197 std::map<Tenum, Tval> readEnumFile(const std::string &filename) {
198 std::map<Tenum, Tval> map;
199 std::vector<std::string> tempVec;
200 std::string comment = "#";
201 std::string input;
202 std::ifstream inputfile(filename);
203 Tenum key;
204 Tval val;
205 if(inputfile.fail()) {
206 std::string errmsg = "Could not open " + filename + " for reading";
207 throw std::runtime_error(errmsg);
208 }
209 int count = 0;
210 while (std::getline(inputfile, input)) {
211 if (input.empty() || (input.rfind(comment, 0) == 0))
212 continue;
213 tempVec = str_split_whitespace(input);
214 key = enum_string2key<Tenum>(tempVec[0]);
215 std::stringstream ss(tempVec[1]);
216 ss >> val;
217 count++;
218 map.insert(std::pair<Tenum, Tval>(key, val));
219 }
220 return map;
221 }
