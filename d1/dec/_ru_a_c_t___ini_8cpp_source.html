44void RuACT::_initOrig(Variables *theVars, RuACTCondition* RuACT_con) {
45 RuACT::setEPS_connect(theVars->RuACT_EPS_com);
46 RuACT::activase = 80. * theVars->RuACTRatio[10];
47
48 // The rate constant used in the model
49 theVars->RuACT_RC.k1 = 0.006 * theVars->RuACTRatio[0]; // The rate constant of the activation of the Rubisco bound with RuBP. This step is associated with the ARubisco theVars->activase content or activity;
50 theVars->RuACT_RC.kn1 = 1.6 * pow(10., -3.) * theVars->RuACTRatio[1]; // The rate constant of E inactivation by binding of RuBP;
51 theVars->RuACT_RC.km1 = 20 * pow(10., -6.) * theVars->RuACTRatio[2]; // The michaelis menton constant for RuBP with E.
52 theVars->RuACT_RC.Ke2 = 0.1 * theVars->RuACTRatio[3]; // Data from Mate et al 1996. Unit: micormolar;
53 theVars->RuACT_RC.Ke3 = 1.600 * theVars->RuACTRatio[4]; // Data from Mate et al 1996. Unit: micormolar;
54 theVars->RuACT_RC.k6 = 2.5 * theVars->RuACTRatio[5] * 4. / 3.; // micromolar per meter square per second, transfered to unit
55 theVars->RuACT_RC.kc = 0.016 * theVars->RuACTRatio[6]; // Michaelis menton constant for CO2
56 theVars->RuACT_RC.ko = 0.448 * theVars->RuACTRatio[7]; // Michaelis menton constant for O2
57 theVars->RuACT_RC.k7 = theVars->RuACT_RC.k6 * 10 * theVars->RuACTRatio[8]; // The rate constant for ecm to ecmr
58 theVars->RuACT_RC.kr = 20. * pow(10., -3.) * theVars->RuACTRatio[9]; // The apparaent michaelis menton constant for RuBP
59
60 //factor = 0.224/0.3;
61 const double factor = 1;
62 RuACT::factor = factor;
63
64 // Assign value to a variable that is transferred to the program
65 RuACTCondition* RuACT_Con = new RuACTCondition();
66 RuACT_Con->ER = 0.05 * 4. * factor; // The concentration of inactive ER
67 RuACT_Con->Eaf = 0.05 * 4. * factor; // The total concentration of E, EC, AND ECM
68 RuACT_Con->ECMR = 0.2 * 4. * factor; // The concentration of ECMR
69 RuACT_Con->RuBP = 2.; // The concentration of ECMR
70
71 theVars->RuACT_Pool.ET = 0.3 * 4. * factor * theVars->RuACTRatio[11]; // The total concentraiton of E, ER, EC, ECM, ECMR , mM;
72 theVars->RuACT_Pool.Rac = 0.0056 * theVars->RuACTRatio[12]; // The concentration of the activase, mM
73 theVars->RuACT_Pool.C = 0.012 * theVars->RuACTRatio[13]; // mM
74 theVars->RuACT_Pool.O = 0.260 * theVars->RuACTRatio[14]; // mM
75 theVars->RuACT_Pool.M = 5. * theVars->RuACTRatio[15];
76}

78void RuACT::_initCalc(Variables *theVars, RuACTCondition* RuACT_Con) {
79 RuACT::setEPS_connect(theVars->RuACT_EPS_com);
80 RuACT::activase *= theVars->RuACTRatio[10];
81
82 // The rate constant used in the model
83 theVars->RuACT_RC.k1 *= theVars->RuACTRatio[0]; // The rate constant of the activation of the Rubisco bound with RuBP. This step is associated with the ARubisco theVars->activase content or activity;
84 theVars->RuACT_RC.kn1 *= theVars->RuACTRatio[1]; // The rate constant of E inactivation by binding of RuBP;
85 theVars->RuACT_RC.km1 *= theVars->RuACTRatio[2]; // The michaelis menton constant for RuBP with E.
86 theVars->RuACT_RC.Ke2 *= theVars->RuACTRatio[3]; // Data from Mate et al 1996. Unit: micormolar;
87 theVars->RuACT_RC.Ke3 *= theVars->RuACTRatio[4]; // Data from Mate et al 1996. Unit: micormolar;
88 theVars->RuACT_RC.k6 *= theVars->RuACTRatio[5] * 4. / 3.; // micromolar per meter square per second, transfered to unit
89 theVars->RuACT_RC.kc *= theVars->RuACTRatio[6]; // Michaelis menton constant for CO2
90 theVars->RuACT_RC.ko *= theVars->RuACTRatio[7]; // Michaelis menton constant for O2
91 theVars->RuACT_RC.k7 *= theVars->RuACTRatio[8]; // The rate constant for ecm to ecmr
92 theVars->RuACT_RC.kr *= theVars->RuACTRatio[9]; // The apparaent michaelis menton constant for RuBP
93
94 //factor = 0.224/0.3;
95 const double factor = RuACT::factor;
96
97 RuACT_Con->ER *= factor; // The concentration of inactive ER
98 RuACT_Con->Eaf *= factor; // The total concentration of E, EC, AND ECM
99 RuACT_Con->ECMR *= factor; // The concentration of ECMR
100
101 theVars->RuACT_Pool.ET *= factor * theVars->RuACTRatio[11]; // The total concentraiton of E, ER, EC, ECM, ECMR , mM;
102 theVars->RuACT_Pool.Rac *= theVars->RuACTRatio[12]; // The concentration of the activase, mM
103 theVars->RuACT_Pool.C *= theVars->RuACTRatio[13]; // mM
104 theVars->RuACT_Pool.O *= theVars->RuACTRatio[14]; // mM
105 theVars->RuACT_Pool.M *= theVars->RuACTRatio[15];
106}
