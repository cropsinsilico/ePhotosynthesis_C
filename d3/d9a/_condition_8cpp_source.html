141void ePhotosynthesis::readFile(const std::string &filename, std::map<std::string, std::string> &mapper) {
142 std::vector<std::string> tempVec;
143 std::string input;
144 std::ifstream inputfile(filename);
145 typename std::map<std::string, std::string>::iterator existing;
146 if(inputfile.fail()) {
147 std::string errmsg = "Could not open " + filename + " for reading";
148 throw std::runtime_error(errmsg);
149 }
150 while (std::getline(inputfile, input)) {
151 if (input.empty())
152 continue;
153 tempVec = utils::str_split_whitespace(input);
154 existing = mapper.find(tempVec[0]);
155 if (existing != mapper.end()) {
156 std::cerr << "readFile[" << filename << "]: " << tempVec[0] <<
157 " already set. Using previous value." << std::endl;
158 continue;
159 }
160 mapper.insert(std::pair<std::string, std::string>(tempVec[0], tempVec[1]));
161 }
162}

164void ePhotosynthesis::readFile(const std::string &filename, std::map<std::string, double> &mapper, bool enzymes) {
165 std::vector<std::string> tempVec;
166 std::string input;
167 std::ifstream inputfile(filename);
168 typename std::map<std::string, double>::iterator existing;
169 if(inputfile.fail()) {
170 std::string errmsg = "Could not open " + filename + " for reading";
171 throw std::runtime_error(errmsg);
172 }
173 int count = 0;
174 while (std::getline(inputfile, input)) {
175 if (input.empty())
176 continue;
177 tempVec = utils::str_split_whitespace(input);
178 double d;
179 std::stringstream ss(tempVec[1]);
180 ss >> d;
181 if (enzymes && (count < 27))
182 d /= 30.;
183 count++;
184 existing = mapper.find(tempVec[0]);
185 if (existing != mapper.end()) {
186 std::cerr << "readFile[" << filename << "]: " << tempVec[0] <<
187 " already set. Using previous value." << std::endl;
188 continue;
189 }
190 mapper.insert(std::pair<std::string, double>(tempVec[0], d));
191 }
