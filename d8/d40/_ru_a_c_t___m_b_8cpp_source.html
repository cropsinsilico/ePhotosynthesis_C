35RuACTCondition* RuACT::_MB_con(const double t, const RuACTCondition* const RuACT_Con, Variables *theVars) {
36
37 Condition(t, theVars);
38 const double light = theVars->GLight;
39
40 theVars->RuACT_Param[0] = light;
41
42 Rate(t, RuACT_Con, theVars);
43
44 const double v1 = theVars->RuACT_Vel.v1; // v1 The rate of ER activation due to Rubisco activase
45 const double vn1 = theVars->RuACT_Vel.vn1; // vn1 The rate of E inactiavtion due to binding of RuBP
46 const double v7 = theVars->RuACT_Vel.v7; // v7 The rate of formation of ECMR from ECM by binding of RuBP
47 const double vn7 = theVars->RuACT_Vel.vn7; // vn7 The rate of actiavtion of ECMR by Rubisco activase
48 const double v6_1 = theVars->RuACT_Vel.v6_1; // v6_1 The rate of RuBP carboxylation
49 const double v6_2 = theVars->RuACT_Vel.v6_2; // v6_2 The rate of RuBP oxygenation
50
51 RuACTCondition* dydt = new RuACTCondition();
52 dydt->ER = vn1 - v1; // ER
53 dydt->Eaf = v1 - v7 + vn7 + v6_1 + v6_2 - vn1; // EAF
54 dydt->ECMR = v7 - vn7 - v6_1 - v6_2; // ECMR
55 if (!theVars->RuACT_EPS_com)
56 dydt->RuBP = v6_1 + v6_2 + v1 - vn1 + vn7 - v7; // RuBP
57#ifdef INCDEBUG
58 DEBUG_DELTA(dydt)
59#endif
60 return dydt;
61}
