45 double NADPH;
46 if (theVars->FIBF_PSPR_com) {

48 } else {

52 // Assuming that the regulation exists no matter there is enzyme regulation or not. ATPReg is
53 // used to regulate the TP export and starch synthesis.

56 const double DHAP = PS_con->T3P / (1. + PS::KE4);
57 const double GAP = PS::KE4 * PS_con->T3P / (1. + PS::KE4);
59 // DHAP= T3P*PS::KE4/(1+PS::KE4); ////WY201803
60 // GAP = T3P/(1+PS::KE4);

63
64 theVars->ADP = PS_C_CA - PS_con->ATP;
65 const double F6P = (PS_con->HexP / PS::KE21) / PS::KE2Ratio;
66 const double G6P = PS_con->HexP / PS::KE2Ratio;
67 const double G1P = (PS_con->HexP * PS::KE22) / PS::KE2Ratio;
68 const double Ru5P = PS_con->PenP / PS::KE1Ratio;
69 const double Ri5P = (PS_con->PenP / PS::KE11) / PS::KE1Ratio;
70 const double Xu5P = (PS_con->PenP / PS::KE12) / PS::KE1Ratio;
71
72 double ATPreg;

74 if (!theVars->RedoxReg_RA_com) {
75 ATPreg = PS_con->PGA / 3.;
76 } else {
77 ATPreg = 1.;

80 //theVars->PS_Vel.v1 = v1;
81 if (theVars->useC3) {

84 PS::PiTc = theVars->SUCS_Pool.PTc - 2. * (SUCS_con->FBPc + SUCS_con->F26BPc) -

89
90 theVars->Pi = PS::PS_C_CP - PS_con->PGA - 2. * PS_con->DPGA - GAP - DHAP - 2 * PS_con->FBP -
91 F6P - PS_con->E4P - 2. * PS_con->SBP - PS_con->S7P - Xu5P - Ri5P - Ru5P - 2. *
92 PS_con->RuBP - G6P - G1P - PS_con->ATP - PS_con->parent->PR_con->PGCA;
93
94 // First here is one way of the redox regulation, assuming the regulation is instataneous.
95 // in case that there are more work using the equilibrium of Thio with enzyme
96 // as a way to regulate enzyme activities.
97 if (theVars->RedoxReg_RA_com) {
98 PS::PsV6 = RedoxRegCondition::getV6();
99 PS::PsV9 = RedoxRegCondition::getV9();
100 PS::PsV13 = RedoxRegCondition::getV13();
101 PS::PsV16 = RedoxRegCondition::getV16();
102 }
103
104 const double CO2 = theVars->CO2_cond;
105 const double O2 = theVars->O2_cond;
106
107
108 V1Reg = 1. + PS_con->PGA / PS::KI11 + PS_con->FBP / PS::KI12 + PS_con->SBP / PS::KI13 +
109 theVars->Pi / PS::KI14 + NADPH / PS::KI15; // SHARED
110
111 if (theVars->RUBISCOMETHOD == 2) {
112 const double tmp = PS::PsV1 * PS_con->RuBP / (PS_con->RuBP + PS::KM13 * PS::V1Reg);
113 theVars->PS_Vel.v1 = tmp * CO2 / (CO2 + PS::KM11 * (1. + O2 / PS::KM12));
114 if (PS_con->RuBP < PS::PsV1 / 2.) {
115 theVars->PS_Vel.v1 = theVars->PS_Vel.v1 * PS_con->RuBP / (PS::PsV1 / 2.);
116 }
117
118 } else if (theVars->RUBISCOMETHOD == 1) {
119 theVars->PS_Vel.v1 = PS::PsV1 * CO2 / (CO2 + KM11 * (1. + O2 / PS::KM12));
120 if (PS_con->RuBP < PS::PsV1 / 2.)
121 theVars->PS_Vel.v1 = theVars->PS_Vel.v1 * PS_con->RuBP / (PS::PsV1 / 2.);
122 }
123
124 theVars->PS_Vel.v2 = PS::PsV2 * PS_con->PGA * PS_con->ATP / ((PS_con->PGA + PS::KM21) *
125 (PS_con->ATP + PS::KM22 *
126 (1. + theVars->ADP / PS::KM23)));
127 theVars->PS_Vel.v3 = PS::PsV3 * PS_con->DPGA * NADPH / ((PS_con->DPGA + PS::KM31a) *
128 (NADPH + PS::KM32b));
129 theVars->PS_Vel.v5 = PS::PsV5 * (GAP * DHAP - PS_con->FBP / PS::KE5) /
130 ((PS::KM51 * PS::KM52) * (1. + GAP / PS::KM51 + DHAP / PS::KM52 +
131 PS_con->FBP / PS::KM53 + GAP * DHAP /
132 (PS::KM51 * PS::KM52)));
133 theVars->PS_Vel.v8 = PS::PsV8 * (DHAP * PS_con->E4P - PS_con->SBP / PS::KE8) /
134 ((PS_con->E4P + PS::KM82) * (DHAP + PS::KM81));
135 theVars->PS_Vel.v6 = PS::PsV6 * (PS_con->FBP - F6P * theVars->Pi / PS::KE6) /
136 (PS_con->FBP + PS::KM61 * (1. + F6P / PS::KI61 + theVars->Pi / PS::KI62));
137 theVars->PS_Vel.v7 = PS::PsV7 * (F6P * GAP - Xu5P * PS_con->E4P / PS::KE7) /
138 ((F6P + PS::KM73 * (1. + Xu5P / PS::KM71 + PS_con->E4P / PS::KM72))
139 * (GAP + KM74));
140 theVars->PS_Vel.v9 = PS::PsV9 * (PS_con->SBP - theVars->Pi * PS_con->S7P / PS::KE9) /
141 (PS_con->SBP + PS::KM9 * (1. + theVars->Pi / PS::KI9));
142 theVars->PS_Vel.v10 = PS::PsV10 * (GAP * PS_con->S7P - Ri5P * Xu5P / PS::KE10) /
143 ((GAP + PS::KM102 * (1. + Xu5P / PS::KM101 + Ri5P / PS::KM10)) *
144 (PS_con->S7P + PS::KM103));
145 theVars->PS_Vel.v13 = PS::PsV13 * (PS_con->ATP * Ru5P - theVars->ADP * PS_con->RuBP / PS::KE13) /
146 ((PS_con->ATP * (1. + theVars->ADP / PS::KI134) + PS::KM132 *
147 (1. + theVars->ADP / PS::KI135)) * (Ru5P + PS::KM131 *
148 (1. + PS_con->PGA / PS::KI131 +
149 PS_con->RuBP / PS::KI132 +
150 theVars->Pi / PS::KI133)));
151
152 theVars->PS_Vel.v16 = std::min(PS::beta * PS::J, PS::PsV16 * (theVars->ADP * theVars->Pi -
153 PS_con->ATP / PS::KE16) /
154 (PS::KM161 * PS::KM162 * (1. + theVars->ADP / PS::KM161 +
155 theVars->Pi / PS::KM162 +
156 PS_con->ATP / PS::KM163 +
157 theVars->ADP * theVars->Pi /
158 (PS::KM161 * PS::KM162))));
159
160 theVars->PS_Vel.v23 = PS::PsV23 * G1P * PS_con->ATP /((G1P + PS::KM231) *
161 ((1. + theVars->ADP / PS::KI23) *
162 (PS_con->ATP + PS::KM232) +
163 (PS::KM232 * theVars->Pi /
164 (PS::KA231 * PS_con->PGA + PS::KA232 *
165 F6P + PS::KA233 * PS_con->FBP))));
166
167 theVars->PS_Vel.v23 = PS::PsV23 * G1P * PS_con->ATP / ((G1P + PS::KM231) *
168 ((1. + theVars->ADP / PS::KI23) *
169 (PS_con->ATP + PS::KM232) +
170 (PS::KM232 * theVars->Pi /
171 (PS::KA231 * PS_con->PGA +
172 PS::KA232 * F6P + PS::KA233 *
173 PS_con->FBP))));
174 const double N = 1. + (1. + PS::KM313 / PsPEXT) * (theVars->Pi / PS::KM312 + PS_con->PGA /
175 PS::KM32 + GAP / PS::KM33 + DHAP / PS::KM311);
176
177 // The ATP regualtion really is implicit in the light regulation of sucrose synthesis.
178 theVars->PS_Vel.v31 = PS::PsV31 * DHAP / (N * PS::KM311) ;
179 theVars->PS_Vel.v32 = PS::PsV32 * PS_con->PGA / (N * PS::KM32);
180 theVars->PS_Vel.v33 = PS::PsV33 * GAP / (N * PS::KM33);
181
182 theVars->PS_Vel.v23 = theVars->PS_Vel.v23 * ATPreg;
183 theVars->PS_Vel.v31 = theVars->PS_Vel.v31 * ATPreg;
184 theVars->PS_Vel.v32 = theVars->PS_Vel.v32 * ATPreg;
185 theVars->PS_Vel.v33 = theVars->PS_Vel.v33 * ATPreg;
186 if (!theVars->FIBF_PSPR_com){ // ModelMethod = 0 means that there is no connection between FIBF and PSPR.
187 if (theVars->PS_Vel.v16 == 0.) // This assmed that light reguate the export of triose phosphate export. This function should use
188 theVars->PS_Vel.v23 = 0.; // ATP as a signal.
189 } else {
190 if (BF::getEPS_ATP_Rate() == 0.)
191 theVars->PS_Vel.v23 = 0;
192 }
193 // Notice the series PS2CM is used both in the CM model and the FPSReg model and thereafter.
194 //theVars->EPS_NADPH = PS_con->NADPH;
195
196 } else {
197 const double Pit = PS::PS_C_CP - PS_con->PGA - 2. * PS_con->DPGA - GAP - DHAP - 2. *
198 PS_con->FBP - F6P - PS_con->E4P - 2. * PS_con->SBP - PS_con->S7P - Xu5P -
199 Ri5P - Ru5P - 2. * PS_con->RuBP - G6P - G1P - PS_con->ATP - PS::Param[1];
200 theVars->Pi = 0.5 * (-PS::KE25 + pow((PS::KE25 * PS::KE25 + 4. * Pit * PS::KE25), 0.5));
201 const double OPOP = Pit - theVars->Pi;
202
203 const double Den = 1. + (1. + GAP / PS::Km5gap) * (F6P / PS::Km8f6p + PS_con->S7P / PS::Km8s7p) +
204 GAP / PS::Km8gap + 1. / PS::Km8p5p * (Xu5P * (1. + PS_con->E4P * Ri5P / PS::Km5p5p) +
205 PS_con->E4P + Ri5P);
206
207 const double Va = PS::KVmo + PS::V23 * (PS_con->PGA / (PS::KA231 * (1. + PS_con->PGA / PS::KA231)));
208 const double v23num = Va * (PS_con->ATP * G1P - PS_con->ADPG * OPOP / PS::KE23);// The reason we set this here is to assume that we can obtain a reverse reaction here. However, a more realistic
209 // way to achieve the homeostasis might be to allow starch breakdown and allow regulation of SBPase and FBPase.
210
211 // WY 201803
212 const double v23den2 = (1. + theVars->Pi / PS::KI231) * PS::KM231 * PS::KM232 *
213 (1. + PS_con->ATP / PS::KM232 + G1P / PS::KM231 + PS_con->ATP * G1P /
214 (PS::KM231 * PS::KM232) + PS_con->ADPG / PS::KM233 + OPOP / PS::KM234 +
215 PS_con->ADPG * OPOP / (PS::KM233 * PS::KM234));
216
217 PS::V1Reg = 1. + PS_con->PGA / PS::KI11 + PS_con->FBP / PS::KI12 + PS_con->SBP / PS::KI13 +
218 theVars->Pi / PS::KI14 + NADPH / PS::KI15;
219
220 // Initialize the PrVmax of the different reactions based on the global variables Vmax
221 PS::PsV6 = PS::V6; // 6 FBPase FBP<->F6P+OP
222 PS::PsV9 = PS::V9; // 9 SBPase SBP<->S7P+OP
223 PS::PsV13 = PS::V13; // 13 Ribulosebiphosphate kinase Ru5P+ATP<->RuBP+ADP
224 PS::PsV16 = PS::V16; // 16 ATP synthase ADP+Pi<->ATP
225
226 // First here is one way of the redox regulation, assuming the regulation is instataneous.
227 // in case that there are more work using the equilibrium of Thio with enzyme
228 // as a way to regulate enzyme activities.
229 if (theVars->RedoxReg_RA_com) {
230 PS::PsV6 = RedoxRegCondition::getV6();
231 PS::PsV9 = RedoxRegCondition::getV9();
232 PS::PsV13 = RedoxRegCondition::getV13();
233 PS::PsV16 = RedoxRegCondition::getV16();
234 }
235
236 if (theVars->RUBISCOMETHOD == 2) {
237 const double tmp = V1 * PS_con->RuBP / (PS_con->RuBP + PS::KM13 * PS::V1Reg);
238 theVars->PS_Vel.v1 = tmp * theVars->CO2_cond / (theVars->CO2_cond + PS::KM11 *
239 (1. + theVars->O2_cond / PS::KM12));
240
241 if (PS_con->RuBP < PS::V1 / 2.5)
242 theVars->PS_Vel.v1 = theVars->PS_Vel.v1 * PS_con->RuBP / (PS::V1 / 2.5);
243
244
245 } else if (theVars->RUBISCOMETHOD == 1){
246 //fprintf("M1 ");
247 theVars->PS_Vel.v1 = PS::V1 * theVars->CO2_cond / (theVars->CO2_cond + PS::KM11 *
248 (1. + theVars->O2_cond / PS::KM12));
249 if (PS_con->RuBP < PS::V1 / 2.5)
250 theVars->PS_Vel.v1 = theVars->PS_Vel.v1 * PS_con->RuBP / (PS::V1 / 2.0);// DNF was 2.5 not 2.0
251 }
252
253 theVars->PS_Vel.v2 = PS::V2 * PS_con->PGA * PS_con->ATP / ((PS_con->PGA + PS::KM21) *
254 (PS_con->ATP + PS::KM22 *
255 (1. + theVars->ADP / PS::KM23)));
256 theVars->PS_Vel.v3 = PS::V3 * PS_con->DPGA * NADPH / ((PS_con->DPGA + PS::KM31a) *
257 (NADPH + PS::KM32b));
258 theVars->PS_Vel.v4 = 0.;
259 theVars->PS_Vel.v5 = PS::V5 * (GAP * DHAP - PS_con->FBP / PS::KE5) / ((PS::KM51 * PS::KM52) *
260 (1. + GAP / PS::KM51 +
261 DHAP / PS::KM52 +
262 PS_con->FBP / PS::KM53 +
263 GAP * DHAP /
264 (PS::KM51 * PS::KM52)));
265 theVars->PS_Vel.v6 = PS::PsV6 * (PS_con->FBP - F6P * theVars->Pi / PS::KE6) /
266 (PS_con->FBP + PS::KM61 * (1. + F6P / PS::KI61 + theVars->Pi / PS::KI62));
267
268 theVars->PS_Vel.v7 = PS::V7 * (F6P * GAP * PS::KE57 - PS_con->E4P * Xu5P) /
269 (PS::Km8p5p * PS::Km5p5p * Den);
270 theVars->PS_Vel.v8 = PS::V8 * (DHAP * PS_con->E4P - PS_con->SBP / PS::KE8) /
271 ((PS_con->E4P + PS::KM82) * (DHAP + PS::KM81));
272 theVars->PS_Vel.v9 = PS::PsV9 * (PS_con->SBP - theVars->Pi * PS_con->S7P / PS::KE9) /
273 (PS_con->SBP + PS::KM9 * (1. + theVars->Pi / PS::KI9));
274 theVars->PS_Vel.v10 = PS::V7 * (PS_con->S7P * GAP * PS::KE810 - Xu5P * Ri5P) /
275 (PS::Km8p5p * PS::Km5p5p * Den);
276
277 theVars->PS_Vel.v13 = PS::PsV13 * (PS_con->ATP * Ru5P - theVars->ADP * PS_con->RuBP / PS::KE13) /
278 ((PS_con->ATP * (1. + theVars->ADP / PS::KI134) + PS::KM132 *
279 (1. + theVars->ADP / PS::KI135)) * (Ru5P + PS::KM131 *
280 (1. + PS_con->PGA / PS::KI131 +
281 PS_con->RuBP / PS::KI132 +
282 theVars->Pi / PS::KI133)));
283 theVars->PS_Vel.v16 = PS::PsV16 * (theVars->ADP * theVars->Pi - PS_con->ATP / PS::KE16) /
284 (PS::KM161 * PS::KM162 * (1. + theVars->ADP / PS::KM161 + theVars->Pi /
285 PS::KM162 + PS_con->ATP / PS::KM163 +
286 theVars->ADP * theVars->Pi /
287 (PS::KM161 * PS::KM162)));
288 theVars->PS_Vel.v23 = v23num / v23den2;
289 theVars->PS_Vel.v31 = (PS::PsV31 * DHAP / (DHAP + PS::KM311) * PsPEXT /
290 (PsPEXT + PS::KM313)) * ATPreg;
291 theVars->PS_Vel.v32 = (PS::PsV32 * PS_con->PGA / (PS_con->PGA + PS::KM32) *
292 PsPEXT / (PsPEXT + PS::KM313)) * ATPreg;
293 theVars->PS_Vel.v33 = (PS::PsV33 * GAP / (GAP + PS::KM33) * PsPEXT / (PsPEXT + PS::KM313)) * ATPreg;
294 theVars->PS_Vel.v24 = (PS::V24 * (PS_con->ADPG)) / (PS::KM241 * (1 + PS_con->ADPG / PS::KM241));
295 theVars->PS_Vel.v25 = (0.5 * theVars->PSRatio[102] / 100. / 5.) *
296 (1. - PS_con->RuBP / PS::MaxCoeff) * PS_con->ATP / (PS_con->ATP + 1.);
297 }
298 theVars->PS2BF_Pi = theVars->Pi;
299#ifdef INCDEBUG
300 DEBUG_INTERNAL(theVars->PS_Vel)

304
305 if (theVars->record) {

307 PS::N++;

312 // Transfer the variables for output
313
314 theVars->PS2OUT.RuBP = PS_con->RuBP;
315 theVars->PS2OUT.PGA = PS_con->PGA;
316 theVars->PS2OUT.DPGA = PS_con->DPGA;
317 theVars->PS2OUT.T3P = PS_con->T3P;
318 theVars->PS2OUT.ADPG = PS_con->ADPG;
