59void CMDriver::setup() {
60
62 // Global variables used for obtaining flux and concentration data //
64
66 // Initialation step //
68 IniModelCom(inputVars); // Initialize the structure of the model, i.e. Is this model separate or combined with others.
69 // This is a variable indicating whether the PR model is actually need to be combined with PS or not. If 1 then means combined; 0 means not.
70 inputVars->PR_PS_com = true;
71 // This is a variable indicating whether the PSPR model is actually need to be combined with SUCS or not. If 1 then means combined; 0 means not.
72 inputVars->PSPR_SUCS_com = true;
73
74 CMCondition* CM_con = CM_Ini();

76 delete CM_con;
78 // Calculation step //
80
81 inputVars->SUCS_Param[0] = 1;
82 inputVars->SUCS_Param[1] = 1;
83
84 inputVars->PS_PR_Param = 0;
