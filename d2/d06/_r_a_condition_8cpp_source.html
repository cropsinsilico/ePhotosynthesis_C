73void RACondition::_fromArray(const arr &vec, const std::size_t offset) {
74 if (EPS_con == nullptr)
75 EPS_con = new EPSCondition(this);
76 if (RuACT_con == nullptr)
77 RuACT_con = new RuACTCondition(this);
78 EPS_con->fromArray(vec, offset);
79 RuACT_con->fromArray(vec, offset + EPSCondition::size());
80}
