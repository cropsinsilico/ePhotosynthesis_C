44template<typename ValueType>
45void updateParamYgg(const ValueType& v, Variables* theVars) {
46 std::map<std::string, std::string> out;
47 for (typename ValueType::ConstMemberIterator it = v.MemberBegin();
48 it != v.MemberEnd(); it++) {
49 std::string k(it->name.GetString());
50 std::ostringstream oss;
51 oss << it->value;
52 out[k] = oss.str();
53 }
54 Variables::updateParam(out, theVars, "updateParamYgg");
55}
56template<typename T>
57void param2ygg(const std::map<std::string, T>& v,
58 rapidjson::Document& out,
59 const Variables* theVars) {
60 out.SetObject();
61 for (typename std::map<std::string, T>::const_iterator it = v.begin();
62 it != v.end(); it++) {
63 rapidjson::Value key(it->first.c_str(), it->first.size(),
64 out.GetAllocator());
65 rapidjson::Value val(it->second, out.GetAllocator());
66 out.AddMember(key, val, out.GetAllocator());
67 }
68 out.AddMember("Light intensity", theVars->TestLi, out.GetAllocator());
69}

73ePhotosynthesis::run_simulation(Variables*& theVars,
74 const DriverType driverChoice,
75 const double& begintime,
76 const double& stoptime,
77 const double& stepsize,
78 const int& maxSubSteps,
79 const double& abstol,
80 const double& reltol,
81 const std::string& outputFile,
82 const std::vector<std::string>& outputVars,
83 int outputParam,
84 const std::string& outputParamBase,
85 const std::vector<std::string>& outputParamVars) {
86
87 drivers::Driver *maindriver = nullptr;
88 std::map<std::string, double> output;
89
90#ifdef WITH_YGGDRASIL
91 int flag = 0;
92 Variables* origVars = nullptr;
93 YggInput* steps = nullptr;
94 YggOutput* out = nullptr;
95 bool first = true;
96 int iteration = 0;
97 if (std::getenv("YGG_SUBPROCESS")) {
98 try {
99 steps = new YggInput("param");
100 out = new YggOutput("output");
101 origVars = theVars->deepcopy();
102 } catch(...) {
103 if (steps) {
104 delete steps;
105 steps = nullptr;
106 }
107 }
108 }
109
110 while (true) {
111 if (steps) {
112 std::cout << "ITERATION: " << iteration << std::endl;
113 rapidjson::Document new_state;
114 flag = steps->recvVar(new_state);
115 if (flag < 0) {
116 if (first) {
117 delete theVars;
118 theVars = nullptr;
119 throw std::runtime_error("Failed to receive first step.");
120 } else {
121 std::cout << "End of input" << std::endl;
122 break;
123 }
124 }
125 if (!first) {
126 delete theVars;
127 theVars = origVars->deepcopy();
128 }
129
130 std::cout << "INPUT STATE = " << new_state << std::endl;
131 updateParamYgg(new_state, theVars);
132 } else if (!first) {
133 break;
134 }
135#endif
136
137 maindriver = drivers::create_driver(driverChoice, theVars,
138 begintime, stepsize,
139 stoptime, maxSubSteps,
140 abstol, reltol, 1, 1);
141 if ((outputParam == 0) && !(outputParamBase.empty() &&
142 outputParamVars.empty()))
143 outputParam = 1;
144 if (outputParam) {
145 maindriver->outputParam(static_cast<OutputFreq>(outputParam),
146 outputParamBase,
147 outputParamVars);
148 }
149 if (!outputVars.empty()) {
150 maindriver->setOutputVars(outputVars);
151 }
152 std::vector<double> ResultRate = maindriver->run();
153
154#ifdef WITH_YGGDRASIL
155 if (out) {
156 rapidjson::Document out_state;
157 param2ygg(maindriver->output, out_state, theVars);
158 std::cout << "OUTPUT STATE = " << out_state << std::endl;
159 flag = out->sendVar(out_state);
160 if (flag < 0)
161 throw std::runtime_error("Error sending output.");
162 } else {
163#endif // WITH_YGGDRASIL
164 // std::cerr << theVars << std::endl;
165 if (!outputFile.empty()) {
166 std::ofstream outfile(outputFile);
167 maindriver->writeOutputTable(outfile);
168 outfile.close();
169 }
170#ifdef WITH_YGGDRASIL
171 }
172#endif // WITH_YGGDRASIL
173
174 output = maindriver->getOutput();
175 if (theVars != nullptr) {
176 maindriver->inputVars = nullptr;
177 }
178 delete maindriver;
179
180#ifdef WITH_YGGDRASIL
181 first = false;
182 iteration++;
183 }
184 if (origVars != nullptr)
185 delete origVars;
186#endif
187 return output;
188}

190std::map<std::string, double>
191ePhotosynthesis::run_simulation(DriverType driverChoice,
192 const double& begintime,
193 const double& stoptime,
194 const double& stepsize,
195 const int& maxSubSteps,
196 const double& abstol,
197 const double& reltol,
198 std::map<std::string, std::string> inputs,
199 bool useC3,
200 const std::string& evnFile,
201 const std::string& atpcostFile,
202 const std::string& enzymeFile,
203 const std::string& grnFile,
204 const std::string& outputFile,
205 const std::vector<std::string>& outputVars,
206 const int& outputParam,
207 const std::string& outputParamBase,
208 const std::vector<std::string>& outputParamVars,
209 ushort dbglvl,
210 const bool debugDelta,
211 const bool debugInternal,
212 const bool& record,
213 Variables *theVars) {
214 // TODO: Do yggdrasil initial input before this point so driver
215 // choice can be an input?
216 if (driverChoice == EPS)
217 useC3 = true;
218 // Select driver prior to updating values from user provided files
219 // so that warnings can be generated when a file is provided
220 // that will not be used by the selected driver.
221 drivers::select_driver(driverChoice, useC3);
222
223 bool created_theVars = false;
224 if (!theVars) {
225 theVars = new Variables();
226 created_theVars = true;
227 }
228
229 // Add variables from inputs
230 if (!inputs.empty()) {
231 Variables::updateParam(inputs, theVars,
232 "ePhotosynthesis raw inputs");
233 }
234
235 // Ensure that command line argument takes precedence
236 theVars->readParam(evnFile, inputs);
237 theVars->readParam(atpcostFile, inputs);
238
239 // Read transcription factors and set enzyme activity levels
240 if (!enzymeFile.empty()) {
241 theVars->readEnzymeAct(enzymeFile);
242 } else {
243 if (useC3)
244 throw std::runtime_error("Enzyme data required if --c3 set (automatically true for EPS driver)");
245 }
246
247 // Read the GRN data and assign it into the correct positions
248 // based on the expected order
249 if (!grnFile.empty())
250 theVars->readGRN(grnFile);
251
252 theVars->record = record;
253 theVars->useC3 = useC3; // Redundant after select_driver
254 modules::PR::setRUBISCOTOTAL(3);
255 if (debugDelta)
256 dbglvl += 8;
257 if (debugInternal)
258 dbglvl += 16;
259#ifdef INCDEBUG
260 std::cout << "Setting Debug level " << dbglvl << std::endl;
261 theVars->debuglevel = static_cast<Debug::RequestedDebug>(dbglvl);
262#else
263 if (dbglvl != 0)
264 std::cout << "This is not a debug build, no debug reporting will be done." << std::endl;
265#endif
266
267 std::map<std::string, double> out = run_simulation(
268 theVars, driverChoice,
269 begintime, stoptime, stepsize, maxSubSteps, abstol, reltol,
270 outputFile, outputVars,
271 outputParam, outputParamBase, outputParamVars);
272
273 if (theVars != nullptr)
274 delete theVars;
275 return out;
276}
