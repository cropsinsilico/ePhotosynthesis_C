40void XanCycle::_Rate(const double t, const XanCycleCondition* const XanCycle_Con, Variables *theVars) {
41
42 Condition(t, theVars);
43
44 double pH;
45 if (theVars->XanCycle_BF_com) {
46 pH = XanCycle_Con->parent->RA_con->EPS_con->FIBF_con->BF_con->PHl;
47 } else {
48 pH = 6.;
49 }
50
51 double RegCof;
52 if (pH <= 5.8) {
53 RegCof = 1.;
54 } else if (pH > 5.8 && pH < 6.5){
55 RegCof = (6.5 - pH) / 0.7;
56 } else {
57 RegCof = 0.;
58 }
59
60 theVars->XanCycle_Vel.Vva = XanCycle_Con->Vx * XanCycle::kva * RegCof; // The velocity of v to a conversion
61 theVars->XanCycle_Vel.Vaz = XanCycle_Con->Ax * XanCycle::kaz * RegCof; // The rate of A to z
62 theVars->XanCycle_Vel.Vza = XanCycle_Con->Zx * XanCycle::kza; // THe rate of z to a
63 theVars->XanCycle_Vel.Vav = XanCycle_Con->Ax * XanCycle::kav; // The rate of A to V
64 theVars->XanCycle_Vel.Vvf = VVF; // The rate of V formation
65 theVars->XanCycle_Vel.Vv2ABA = VV2ABA; // The rate of conversion from v to XanCycle_Con.ABA.
66 theVars->XanCycle_Vel.VABAdg = VABADG; // The rate of XanCycle_Con.ABA degradation
67
68#ifdef INCDEBUG
69 DEBUG_INTERNAL(theVars->XanCycle_Vel)
70#endif
71 if (theVars->record) {
72 if (t > XanCycle::TIME) {
73 XanCycle::N++;
74 XanCycle::TIME = t;
75 }
76
77 theVars->XanCycle_VEL.insert(XanCycle::N, t, theVars->XanCycle_Vel);
78
79 theVars->XanCycle2OUT.Vx = XanCycle_Con->Vx;
80 theVars->XanCycle2OUT.Ax = XanCycle_Con->Ax;
81 theVars->XanCycle2OUT.Zx = XanCycle_Con->Zx;
82 theVars->XanCycle2OUT.ABA = XanCycle_Con->Vx / (XanCycle_Con->Vx + XanCycle_Con->Ax +
83 XanCycle_Con->Zx);
84 }
