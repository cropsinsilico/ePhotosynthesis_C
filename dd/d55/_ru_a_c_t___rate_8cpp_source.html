38void RuACT::_Rate(const double t, const RuACTCondition* const RuACT_Con, Variables *theVars) {
39
40 double C = theVars->RuACT_Pool.C;
41 double O = theVars->RuACT_Pool.O;
42 double MT = theVars->RuACT_Pool.M;
43 double RuBP;
44 double ADP;
45 double ATP;
46 if (!theVars->RuACT_EPS_com) {
47 ATP = 1.45;
48 ADP = 1.5 - ATP;
49 RuBP = RuACT_Con->RuBP;
50 } else {
51 // langmm: The top level parent will only have RROEA if it is
52 // trDynaPS, but when the DynaPS driver has RuACT_EPS_com=true,
53 // RROEA will not be present. The result needs to be checked
54 // against the MATLAB code to determine if activase needs
55 // to be calculated differently when running the DynaPS driver.
56 if (theVars->RROEA_EPS_com)
57 RuACT::activase = RuACT_Con->parent->parent->parent->RROEA_con->RuACT * 14364.;
58 RuBP = RuACT_Con->parent->EPS_con->CM_con->PS_PR_con->PS_con->RuBP;
59 C = theVars->CO2_cond;
60 O = theVars->O2_cond;
61
62 MT = RuACT_Con->parent->EPS_con->FIBF_con->BF_con->Mgs;
63 ATP = RuACT_Con->parent->EPS_con->CM_con->PS_PR_con->PS_con->ATP;
64 ADP = PS::getPS_C_CA() - ATP;
65 }
66 double RatioDT = ADP / ATP;
67
68 const double CB = theVars->RuACT_RC.Ke3 + theVars->RuACT_RC.Ke2 * theVars->RuACT_RC.Ke3 / C +
69 RuACT_Con->Eaf - MT;
70 const double CC = - MT * (theVars->RuACT_RC.Ke3 + theVars->RuACT_RC.Ke2 * theVars->RuACT_RC.Ke3 / C);
71
72 const double M = (-CB + pow((pow(CB, 2.) - 4. * CA * CC), 0.5)) / (2. * CA);
73 const double EC = RuACT_Con->Eaf * pow((1 + theVars->RuACT_RC.Ke2 / C + M / theVars->RuACT_RC.Ke3), -1.);
74 const double E = EC / C * theVars->RuACT_RC.Ke2;
75 const double ECM = EC * M / theVars->RuACT_RC.Ke3;
76
77 double LT;
78 double RCA;
79 if (RuACT::activase < pow(10., -6.)) {
80 RCA = 0.;
81 } else {
82 LT = 216.9 / RuACT::activase; // The lifetime of the activation; UNIT: MIN;
83 RCA = 1. / (LT * 60.); // The rate constant of the activation reaction
84 }
85
86 if (RatioDT > 3.) {
87 RatioDT = 3.;
88 } else if (RatioDT < 0.25){
89 RatioDT = 0.;
90 }
91
92 double FATP = 1. - RatioDT / 3.;
93
94 if (FATP < 0.6)
95 FATP = 0.6;
96
97 const double factor_n7 = 1.;
98
99 theVars->RuACT_Vel.v1 = RCA * RuACT_Con->ER * FATP;
100 theVars->RuACT_Vel.vn1 = theVars->RuACT_RC.kn1 * E * RuBP;
101 theVars->RuACT_Vel.v7 = theVars->RuACT_RC.k7 * ECM * RuBP;
102 theVars->RuACT_Vel.vn7 = RuACT_Con->ECMR * 0.5 * factor_n7;
103 theVars->RuACT_Vel.v6_1 = RuACT_Con->ECMR * theVars->RuACT_RC.k6 * C /
104 (C + theVars->RuACT_RC.kc * (1. + O / theVars->RuACT_RC.ko));
105 theVars->RuACT_Vel.v6_2 = RuACT_Con->ECMR * theVars->RuACT_RC.k6 / 3. * O /
106 (O + theVars->RuACT_RC.ko * (1. + C / theVars->RuACT_RC.kc));
107#ifdef INCDEBUG
108 DEBUG_INTERNAL(theVars->RuACT_Vel)
109#endif
110 if (theVars->record) {
111 if (t > RuACT::TIME) {
112 RuACT::N++;
113 RuACT::TIME = t;
114 }
115 theVars->RuACT_VEL.insert(RuACT::N - 1, t, theVars->RuACT_Vel);
116 }
