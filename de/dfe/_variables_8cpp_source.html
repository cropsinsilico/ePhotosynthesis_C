499std::string Variables::parseVar(const std::string& k,
500 MODULE& mod, PARAM_TYPE& pt,
501 const bool& isGlymaID,
502 const bool& use_1st_match,
503 const bool& allow_no_match,
504 bool* controlVar) {
505 mod = MODULE_NONE;
506 pt = PARAM_TYPE_NONE;
507 if (controlVar) controlVar[0] = false;
508 std::string split="::", var1, var2, name;
509 bool var1_used = false, var2_used = false;
510 size_t idx1 = k.find(split);
511 if (idx1 != std::string::npos) {
512 var1 = k.substr(0, idx1);
513 try {
514 mod = utils::enum_string2key<MODULE>(var1);
515 var1_used = true;
516 } catch (...) {
517 try {
518 pt = utils::enum_string2key<PARAM_TYPE>(var1);
519 var1_used = true;
520 } catch (...) {
521 }
522 }
523 size_t idx2 = k.find(split, idx1 + split.size());
524 if (idx2 == std::string::npos) {
525 name = k.substr(idx1 + split.size());
526 } else {
527 var2 = k.substr(idx1 + split.size(),
528 idx2 - (idx1 + split.size()));
529 if (mod == MODULE_NONE) {
530 try {
531 mod = utils::enum_string2key<MODULE>(var2);
532 var2_used = true;
533 } catch (...) {
534 }
535 }
536 if (pt == PARAM_TYPE_NONE && !var2_used) {
537 try {
538 pt = utils::enum_string2key<PARAM_TYPE>(var2);
539 var2_used = true;
540 } catch (...) {
541 }
542 }
543 name = k.substr(idx2 + split.size());
544 }
545 } else {
546 name = k;
547 }
548 if (mod == MODULE_NONE || pt == PARAM_TYPE_NONE) {
549 std::vector<MODULE> check_modules;
550 std::vector<PARAM_TYPE> check_param_types;
551 if (mod != MODULE_NONE)
552 check_modules.push_back(mod);
553 else
554 check_modules = ALL_MODULE;
555 if (pt != PARAM_TYPE_NONE)
556 check_param_types.push_back(pt);
557 else
558 check_param_types = ALL_PARAM_TYPE;
559 std::vector<std::pair<MODULE, PARAM_TYPE> > matches;
560 for (std::vector<MODULE>::const_iterator it_mod = check_modules.begin();
561 it_mod != check_modules.end(); it_mod++) {
562 if (use_1st_match && !matches.empty())
563 break;
564 for (std::vector<PARAM_TYPE>::const_iterator it_pt = check_param_types.begin();
565 it_pt != check_param_types.end(); it_pt++) {
566 if ((((*it_mod) == MODULE_ALL) || ((*it_pt) == PARAM_TYPE_VARS)) &&
567 (((*it_mod) != MODULE_ALL) || ((*it_pt) != PARAM_TYPE_VARS)))
568 continue;
569 if (use_1st_match && !matches.empty())
570 break;
571 if (hasVar(*it_mod, *it_pt, name, isGlymaID, controlVar)) {
572 if ((!(controlVar && controlVar[0])) &&
573 (!isSelected(*it_mod, *it_pt))) {
574 INFO_VALUE_SET("Variable \"", k, "\" matches ",
575 name, " in ",
576 utils::enum_key2string(*it_mod),
577 " ",
578 utils::enum_key2string(*it_pt),
579 " value set, but it is not selected by the current driver");
580 } else {
581 matches.emplace_back(*it_mod, *it_pt);
582 }
583 }
584 }
585 }
586 if (matches.size() == 1) {
587 mod = matches[0].first;
588 pt = matches[0].second;
589 } else if (matches.size() > 1) {
590 std::stringstream ss;
591 for (std::vector<std::pair<MODULE, PARAM_TYPE> >::const_iterator it = matches.begin();
592 it != matches.end(); it++) {
593 if (ss.tellp() > 0)
594 ss << ", ";
595 ss << it->first << "::" << it->second << "::" << name;
596 }
597 ERROR_VALUE_SET("More than one (", matches.size(),
598 ") variable matches \"", k,
599 "\". Use a module & parameter type prefix to disambiguate [matches = ",
600 ss.str(), "]");
601 } else {
602 if (allow_no_match)
603 return "";
604 ERROR_VALUE_SET("Could not find variable matching string \"",
605 k, "\"");
606 }
607 }
608 // INFO_VALUE_SET("parseVar[", k, "] -> ",
609 // utils::enum_key2string(mod), ", ",
610 // utils::enum_key2string(pt), ", ", name);
611 if (controlVar && controlVar[0]) {
612 return getControlAlias(mod, pt, name);
613 }
614 if (mod != MODULE_NONE && pt != PARAM_TYPE_NONE && !name.empty())
615 return GET_VALUE_SET_CLASS(mod, pt)->getAliasedName(name);
616 return name;
617}

1057 MODULE mod = MODULE_NONE;
1058 PARAM_TYPE pt = PARAM_TYPE_NONE;
1059 std::string name, name_SET, name_FULL;
1060 double value = 0.0;
1061 static std::string set_suffix = "_SETXXX";
1062 bool controlVar = false;
1063 std::string controlStr = "";
1064
1065 for (typename std::map<std::string, std::string>::const_iterator it = inputs.begin();
1066 it != inputs.end(); it++) {
1067 if (utils::str_endswith(it->first, set_suffix))
1068 continue;
1069 mod = MODULE_NONE;
1070 pt = PARAM_TYPE_NONE;
1071 controlVar = false;
1072 controlStr = "";

1074 &controlVar);
1075 if (name.empty()) {
1076 std::cout << context << ": IGNORING \"" << it->first <<
1077 "\" - it does not match any known parameters or belongs "
1078 "to a value set not used selected by the current driver."
1079 << std::endl;
1080 rm_values.push_back(it->first);
1081 continue;
1082 }
1083 if (controlVar)
1084 controlStr = "[CONTROL VARIABLE] ";
1085 value = static_cast<double>(std::stof(it->second, nullptr));
1086 name_FULL = utils::enum_key2string(mod) + "::" +
1087 utils::enum_key2string(pt) + "::" + name;
1088 name_SET = name_FULL + set_suffix;
1089 if (inputs.find(name_SET) == inputs.end()) {
1090 std::cout << context << ": READ \"" << name_FULL <<
1091 "\" " << controlStr <<
1092 "from \"" << it->first << "\" (" << it->second << ")" <<
1093 std::endl;
1094 add_values[name_SET] = it->second;
1095 if (theVars) {
1096 if (controlVar)
1097 theVars->setControlVar(mod, pt, name, (int)value);
1098 else
1099 theVars->setVar(mod, pt, name, value);
1100 } else {
1101 if (controlVar)
1102 setDefaultControlVar(mod, pt, name, (int)value);
1103 else
1104 setDefault(mod, pt, name, value);
1105 }
1106 } else {
1107 std::cout << context << ": IGNORING \"" << name_FULL <<
1108 "\" - parameter already set." << std::endl <<
1109 " Current value: " << inputs.at(name_SET) << std::endl <<
1110 " Ignored value: " << it->second << std::endl;
1111 rm_values.push_back(name);
1112 }
1113 rm_values.push_back(it->first);
1114 }
1115 for (typename std::vector<std::string>::iterator it = rm_values.begin();
1116 it != rm_values.end(); it++) {
1117 inputs.erase(*it);
1118 }
1119 inputs.insert(add_values.begin(), add_values.end());
1120}
