55 // Get the mass balance equation //

58 // This page defines the mass balance equation for the system under study
59 // One problem need to be taken care of is the variables needed to transfer from FI_CalV to FI_mb
60 // The Major Variables
61 FICondition* dydt = new FICondition();
62 dydt->A = theVars->FI_Vel.Ia - theVars->FI_Vel.vA_f - theVars->FI_Vel.vA_d -
63 theVars->FI_Vel.vA_U + theVars->FI_Vel.vU_A; // A 6
64 dydt->U = theVars->FI_Vel.Ic + theVars->FI_Vel.vA_U - theVars->FI_Vel.vU_A -
65 theVars->FI_Vel.vU_f - theVars->FI_Vel.vU_d - theVars->FI_Vel.v1 +
66 theVars->FI_Vel.v_r1 - theVars->FI_Vel.vP680qU; // U
67 dydt->P680ePheo = theVars->FI_Vel.vU_P680 + theVars->FI_Vel.v_r1 - theVars->FI_Vel.v1 -
68 theVars->FI_Vel.vP680_d; // P680ePheo QF add
69 dydt->P680pPheon = theVars->FI_Vel.v1 - theVars->FI_Vel.v_r1 - theVars->FI_Vel.vz_1 -
70 theVars->FI_Vel.v2_1 + theVars->FI_Vel.vr2_1; // P680pPheon
71 dydt->P680pPheo = theVars->FI_Vel.v2_1 - theVars->FI_Vel.vr2_1 - theVars->FI_Vel.vz_2; // P680pPheo
72 dydt->P680Pheon = theVars->FI_Vel.vz_1 - theVars->FI_Vel.v2_2 + theVars->FI_Vel.vr2_2; // P680Pheon
73 dydt->Yz = theVars->FI_Vel.vS1_S2 + theVars->FI_Vel.vS2_S3 + theVars->FI_Vel.vS3_S0 +
74 theVars->FI_Vel.vS0_S1 - theVars->FI_Vel.vz_1 - theVars->FI_Vel.vz_2; // Yz
75 dydt->S1T = theVars->FI_Vel.vS0_S1 - theVars->FI_Vel.v1z; // S1T
76 dydt->S2T = theVars->FI_Vel.vS1_S2 - theVars->FI_Vel.v2z; // S2T
77 dydt->S3T = theVars->FI_Vel.vS2_S3 - theVars->FI_Vel.v3z; // S3T
78 dydt->S0T = theVars->FI_Vel.vS3_S0 - theVars->FI_Vel.v0z; // S0T
79 dydt->S1Tp = theVars->FI_Vel.v1z - theVars->FI_Vel.vS1_S2; // S1Tp
80 dydt->S2Tp = theVars->FI_Vel.v2z - theVars->FI_Vel.vS2_S3; // S2Tp
81 dydt->S3Tp = theVars->FI_Vel.v3z - theVars->FI_Vel.vS3_S0; // S3Tp
82 dydt->S0Tp = theVars->FI_Vel.v0z - theVars->FI_Vel.vS0_S1; // S0Tp
83 dydt->QAQB = theVars->FI_Vel.v3 - theVars->FI_Vel.v_r3 - theVars->FI_Vel.v2_00_1 -
84 theVars->FI_Vel.v2_00_2 + theVars->FI_Vel.vr2_00_1 + theVars->FI_Vel.vr2_00_2; // QAQB
85 dydt->QAnQB = theVars->FI_Vel.v2_00_1 + theVars->FI_Vel.v2_00_2 - theVars->FI_Vel.vr2_00_1 -
86 theVars->FI_Vel.vr2_00_2 - theVars->FI_Vel.vAB1 + theVars->FI_Vel.vBA1 + theVars->FI_Vel.v3_n - theVars->FI_Vel.v_r3_n; // QAnQB
87 dydt->QAQBn = theVars->FI_Vel.vAB1 - theVars->FI_Vel.vBA1 - theVars->FI_Vel.v2_01_1 -
88 theVars->FI_Vel.v2_01_2 + theVars->FI_Vel.vr2_01_1 + theVars->FI_Vel.vr2_01_2; // QAQBn
89 dydt->QAnQBn = theVars->FI_Vel.vBA2 - theVars->FI_Vel.vAB2 + theVars->FI_Vel.v2_01_1 +
90 theVars->FI_Vel.v2_01_2 - theVars->FI_Vel.vr2_01_1 - theVars->FI_Vel.vr2_01_2; // QAnQBn
91 dydt->QAQB2n = theVars->FI_Vel.vAB2 - theVars->FI_Vel.vBA2 - theVars->FI_Vel.v3 +
92 theVars->FI_Vel.v_r3 - theVars->FI_Vel.v2_02_1 - theVars->FI_Vel.v2_02_2 +
93 theVars->FI_Vel.vr2_02_1 + theVars->FI_Vel.vr2_02_2; // QAQB2n
94 dydt->QAnQB2n = 0. - theVars->FI_Vel.v3_n + theVars->FI_Vel.v_r3_n + theVars->FI_Vel.v2_02_1 +
95 theVars->FI_Vel.v2_02_2 - theVars->FI_Vel.vr2_02_1 - theVars->FI_Vel.vr2_02_2; // QAnQB2n
96 if (!FICondition::BF_connect)
97 dydt->PQn = theVars->FI_Vel.v3 + theVars->FI_Vel.v3_n - theVars->FI_Vel.v_r3 -
98 theVars->FI_Vel.v_r3_n - theVars->FI_Vel.v_pq_ox; // PQn
99
100#ifdef INCDEBUG
101 DEBUG_DELTA(dydt)
102#endif
103 return dydt;
104}
