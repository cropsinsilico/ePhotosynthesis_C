36RACondition* RA::_MB_con(const double t, const RACondition* const RA_Con, Variables *theVars) {
37
38 EPSCondition* EPSdydt = EPS::MB_con(t, RA_Con->EPS_con, theVars);
39 RuACTCondition* RuACTdydt = RuACT::MB_con(t, RA_Con->RuACT_con, theVars);
40
41 EPSdydt->CM_con->PS_PR_con->PS_con->RuBP = theVars->RuACT_Vel.v1 + theVars->PS_Vel.v13 -
42 theVars->RuACT_Vel.vn1 + theVars->RuACT_Vel.vn7 -
43 theVars->RuACT_Vel.v7;
44
45 EPSdydt->CM_con->PS_PR_con->PS_con->PGA = EPSdydt->CM_con->PS_PR_con->PS_con->PGA -
46 2 * theVars->PS_Vel.v1 + 2 * theVars->RuACT_Vel.v6_1 -
47 theVars->PR_Vel.v111 + theVars->RuACT_Vel.v6_2; // Originally it is pspr(2), now use EPS_DYDT[53].
48
49 EPSdydt->CM_con->PS_PR_con->PR_con->PGCA = EPSdydt->CM_con->PS_PR_con->PR_con->PGCA -
50 theVars->PR_Vel.v111 + theVars->RuACT_Vel.v6_2;
51
52 RACondition* dydt = new RACondition(EPSdydt, RuACTdydt);
53#ifdef INCDEBUG
54 DEBUG_DELTA(dydt)
55#endif
56 return dydt;
57}
