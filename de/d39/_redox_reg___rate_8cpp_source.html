48void RedoxReg::_Rate(const double t, const RedoxRegCondition* const RedoxReg_Con, Variables *theVars) {
49 const double Thio = ThioT - RedoxReg_Con->Thion;
50
51 theVars->RedoxReg_MP[0][2] = RedoxReg_Con->Thion / ThioT;
52
53 double TEMP = theVars->RedoxReg_MP[0][2];
54
55 if (theVars->RROEA_EPS_com)
56 TEMP = 0.5;

58 if (trDynaPS2RedReg_cal) {
59 drivers::UserData *data = drivers::alloc_user_data();
60 data->coeffs.resize(2);
61
62 data->coeffs[0] = theVars->RedoxReg_MP[0][1] - 0.03 * log10(TEMP / (1. - TEMP));
63
64 N_Vector y, constraints, scaling;
65#ifdef SUNDIALS_CONTEXT_REQUIRED
66 y = N_VNew_Serial(1, theVars->context());
67 constraints = N_VNew_Serial(1, theVars->context());
68 scaling = N_VNew_Serial(1, theVars->context());
69#else // SUNDIALS_CONTEXT_REQUIRED
70 y = N_VNew_Serial(1);
71 constraints = N_VNew_Serial(1);
72 scaling = N_VNew_Serial(1);
73#endif // SUNDIALS_CONTEXT_REQUIRED
74 realtype* y_ptr = N_VGetArrayPointer(y);
75 realtype* constraints_ptr = N_VGetArrayPointer(constraints);
76 N_VConst(ONE, scaling);
77 realtype abstol = 1e-5;

81#ifdef SUNDIALS_CONTEXT_REQUIRED

85 SUNLinearSolver LS = SUNLinSol_Dense(y, A, theVars->context());

88
89 SUNMatrix A = SUNDenseMatrix(1, 1);

91#endif // SUNDIALS_CONTEXT_REQUIRED
92#define RUN_CHECK(func, args, success) \
93 { \
94 int error_code = func args; \

103 }
104
105 try {
106 for (std::size_t index = 1; index < 5; index++) {
107
108 y_ptr[0] = theVars->RedoxReg_MP[index][2];
109 data->coeffs[1] = theVars->RedoxReg_MP[index][1];
110 RUN_CHECK(KINInit, (kmem, RedoxReg_FPercent, y), KIN_SUCCESS);

117 if (err > KIN_INITIAL_GUESS_OK) {
118 throw std::runtime_error("KinSol failed: " + std::to_string(err));
119 }
120 realtype *retvals = N_VGetArrayPointer(y);
121 //pr = fsolve(@RedoxReg_FPercent, RedPercent, optimset('Display', 'off'), RedP, MPE);
122 theVars->RedoxReg_MP[index][2] = retvals[0];
123
124
125 if (static_cast<int>(theVars->RedoxReg_MP[index][0]) == 6) {
126 RedoxRegCondition::setV6(RedoxReg_VMAX6 * theVars->RedoxReg_MP[index][2]);
127 } else if (static_cast<int>(theVars->RedoxReg_MP[index][0]) == 9) {
128 RedoxRegCondition::setV9(RedoxReg_VMAX9 * theVars->RedoxReg_MP[index][2]);
129 } else if (static_cast<int>(theVars->RedoxReg_MP[index][0]) == 13) {
130 RedoxRegCondition::setV13(RedoxReg_VMAX13 * theVars->RedoxReg_MP[index][2]);
131 } else if (static_cast<int>(theVars->RedoxReg_MP[index][0]) == 16) {
132 RedoxRegCondition::setV16(RedoxReg_VMAX16 * theVars->RedoxReg_MP[index][2]);
133 }
134 }
135 } catch(...) {
136 std::exception_ptr eptr = std::current_exception();
137 N_VDestroy(y);
138 N_VDestroy(constraints);
139 N_VDestroy(scaling);
140 KINFree(&kmem);
141 SUNLinSolFree(LS);
142 std::rethrow_exception(eptr);
143 }
144 N_VDestroy(y);
145 N_VDestroy(constraints);
146 N_VDestroy(scaling);
147 KINFree(&kmem);
148 SUNLinSolFree(LS);
149 }
150
151 theVars->RedoxReg_Vel.Vred = RedoxReg_Con->RA_con->EPS_con->FIBF_con->BF_con->Fdn * Fd_Thio_ET * Thio / ThioT;
152 theVars->RedoxReg_Vel.Vox = RedoxReg_Con->Thion * Thio_Oxidation;
153
154#ifdef INCDEBUG
155 DEBUG_INTERNAL(theVars->RedoxReg_Vel)
156#endif
157 if (theVars->record) {
158 if (t > RedoxReg::TIME) {
159 RedoxReg::N++;
160 RedoxReg::TIME = t;
161 }
162
163 theVars->RedoxReg_VEL.insert(RedoxReg::N - 1, t, theVars->RedoxReg_Vel);
164 }
