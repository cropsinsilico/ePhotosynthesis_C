53 static bool compareState(const Variables *theVars1,
54 const Variables *theVars2,
55 const U* con1, const U* con2,
56 const std::map<int, double>& map1,
57 const std::map<int, double>& map2) {
58 if ((*theVars1) != (*theVars2)) {
59 std::cerr << "theVars differs" << std::endl;
60 return false;
61 }
62 if ((*con1) != (*con2)) {
63 std::cerr << "con differs" << std::endl;
64 return false;
65 }
66 if (!BaseClass::compare_value_maps(map1, map2)) {
67 std::cerr << "maps differ" << std::endl;
68 return false;
69 }
70 return T::_compareState(theVars1, theVars2, con1, con2, map1, map2);
71 }
72 static std::string diffState(const Variables *theVars1,
73 const Variables *theVars2,
74 const U* con1, const U* con2,
75 const std::map<int, double>& map1,
76 const std::map<int, double>& map2) {
77 std::string out;
78 std::size_t padKeys = std::max(theVars1->max_field_width_all(),
79 theVars2->max_field_width_all());
80 std::size_t padVals = std::max(theVars1->max_value_width_all(),
81 theVars2->max_value_width_all());
82 padVals = std::max(padVals,
83 std::max(con1->max_value_width(),
84 con2->max_value_width()));
85 padVals = std::max(padVals,
86 std::max(BaseClass::value_width_value_map(map1),
87 BaseClass::value_width_value_map(map2)));
88 out += theVars1->diff(*theVars2, padKeys, padVals, true);
89 out += con1->diff(*con2, padKeys, padVals, true);
90 out += BaseClass::diff_value_maps(map1, map2,
91 padKeys, padVals, true);
92 return out;
93 }

113 static U* init(Variables *theVars) {
114#ifdef CHECK_NEW_MATCHES_ORIG
115 bool init0 = initialized();
116 std::map<int, double> map0 = getValueMap();
117 std::map<int, double> preinit_map0 = getValueMap(true);
118 Variables *theVars2 = theVars->deepcopy();
119 if ((*theVars2) != (*theVars)) {
120 ERROR_VALUE_SET("Copied variables does not match\n",
121 theVars->diff(*theVars));
122 }
123 U* out = initOrig(theVars);
124 std::map<int, double> map = getValueMap();
125 ValueSetClass::resetValues(true);
126 setValueMap(map0, init0); // TODO: Set initialized
127 if (!init0)
128 setValueMap(preinit_map0, false, true);
129 U* out2 = initNew(theVars2);
130 std::map<int, double> map2 = getValueMap();
131 if (!compareState(theVars, theVars2, out, out2, map, map2)) {
132 ERROR_VALUE_SET("Old and new methods do not match\n",
133 diffState(theVars, theVars2,
134 out, out2, map, map2));
135 }
136 delete theVars2;
137 delete out2;
138 setValueMap(map, true);
139 return out;
140#else // CHECK_NEW_MATCHES_ORIG
141 return initNew(theVars);
142#endif // CHECK_NEW_MATCHES_ORIG
143 }

154 static U* initAllValues(Variables *theVars,
155 const bool noDefaults=false) {
156 U* out = T::_createCondition(theVars);
157 theVars->initParam(ID, PARAM_TYPE_MOD, noDefaults, true);
158 theVars->initParam(ID, PARAM_TYPE_COND, noDefaults, false,
159 static_cast<ValueSet_t*>(out));
160 const std::vector<PARAM_TYPE>& param_types = ValueSetClass::get_parameter_types();
161 for (typename std::vector<PARAM_TYPE>::const_iterator it = param_types.begin();
162 it != param_types.end(); it++) {
163 if (*it == PARAM_TYPE_COND) continue;
164 theVars->initParam(ID, *it, noDefaults);
165 }
166 return out;
167 }
