41void RROEA::_initOrig(Variables *theVars, RROEACondition* RROEA_con) {
42
43 double Coeff = 1.;
44 theVars->RROEA_RC.Coeff = Coeff;
45
46 // The rate constant used in the model
47 theVars->RROEA_RC.ke2GAPDH = 22. / 60. * Coeff; // The rate constant of electron transfer to GAPDH. From literature.
48 theVars->RROEA_RC.ke2MDH = 20. / 60. * Coeff; // The rate constant of electront transfer to MDH, this rate is totally ASSUMED.
49 theVars->RROEA_RC.ke2FBPase = 1.38 / 60. * Coeff; // The rate constant of electron transfer from thioredoxin to FBPase.
50 theVars->RROEA_RC.ke2SBPase = 1.65 / 60. * Coeff; // The rate constant of electron tranfer from thioredoxin to SBPase
51 theVars->RROEA_RC.ke2PRK = 59.8 / 60. * Coeff; // The rate constant of electron transfer from thioredoxin to PRK, Phosphoribulase kinase
52 theVars->RROEA_RC.ke2RubACT = 6.35 / 60. * Coeff; // The rate constant of electron transfer from thioredoxin to Rubisco activase
53 theVars->RROEA_RC.ke2Fd = 18.5 * Coeff; // The rate constant of electron transfer to fe
54 // Jmax of 180 micro mole per meter square per second.
55 theVars->RROEA_RC.keFd2Thio = 10. * Coeff; // The rate constant of electron transfer from fd to thio
56 theVars->RROEA_RC.keFd2Calvin = 7. * Coeff; // The rate constant of electron transfer from fd to Calvin cycle
57 // Of course, this is a big assumption where the electron is transfered to NADPH
58 // then to the Calvin cycle. This rate is much higher than the electron transfer to differnet
59 // enzymes.
60
61 theVars->RROEA_RC.ke2ATPGPP = 6.3 / 60 * Coeff; // The rate constant of electron transfer to ATPGPP
62 theVars->RROEA_RC.ke2ATPase = 1.0; // The rate constant of electron transfer to ATPase
63
64 // Here is all the equilibriun constants for the different reactions in photosystem
65 theVars->RROEA_KE.KEe2FBPase = 0.311167869;
66 theVars->RROEA_KE.KEe2SBPase = 0.459194309;
67 theVars->RROEA_KE.KEe2PRK = 0.677638775;
68 theVars->RROEA_KE.KEe2ATPase = 1.0; // 2.177727336 was set, but unused, 1.0 is used in RROEA_Rate
69 theVars->RROEA_KE.KEe2RuACT = 0.677638775;
70 theVars->RROEA_KE.KEe2GAPDH = 0.044461692;
71 theVars->RROEA_KE.KEe2MDH = 0.044461692;
72 theVars->RROEA_KE.KEe2ATPGPP = 1.;
73 theVars->RROEA_KE.KEeFd2Thio = 24776.;
74
75 // The following calculate the total concentration of different enzymes.
76 const double MDH_Vmax = 2; // This value is assumed and there is no literature about it. Need to be
77 // fixed.
78 theVars->RROEA_Pool.MDH_Vmax = MDH_Vmax;
79
80 double SC;
81 double fc16;
82 double FC;
83 if (!theVars->RROEA_EPS_com) {
84
85 FC = 1.;
86 fc16 = 1.;
87 SC = 1.;
88 RROEA::FC = FC;
89 RROEA::fc16 = fc16;
90 RROEA::SC = SC;
91
92 PS::setV3(5.04 * SC); // (Harris & Koniger, 1997) 3 GAP dehydragenase DPGA+NADPH <->GAP + OP+NADP
93 PS::setV6(1.155 * SC); // (Harris & Koniger, 1997) 6 FBPase FBP<->F6P+OP 1.155
94 PS::setV9(0.168 * SC * FC); // (Harris & Koniger, 1997) 9 SBPase SBP<->S7P+OP 0.168 as original value; 0.4168 was its value.
95 PS::setV13(8.0094 * SC); // (Harris & Koniger, 1997) 13 Ribulosebiphosphate kinase Ru5P+ATP<->RuBP+ADP
96 PS::setV16(3. * SC * fc16); // (Aflalo & Shavit, 1983, Davenport & McLeod, 1986) 16 ATP synthase ADP+Pi<->ATP 1.47
97 PS::setV23(1.68 * SC * FC); // (Latzko, Steup & Schachtele, 1981) 23 ADP-glucose pyrophosphorylase and ADPG+Gn<->G(n+1)+ADP 0.18
98 }

136 theVars->RROEA_Pool.mw_ATPase = mw_ATPase;

138 if (theVars->RROEA_EPS_com)
139 FdT = theVars->BF_Pool.kU_f;

141 theVars->RROEA_Pool.GAPDH = PS::getV3() * 1000. * 60. / SA_GAPDH / mw_GAPDH;
142 theVars->RROEA_Pool.FBPase = PS::getV6() * 1000. * 60. / SA_FBPase / mw_FBPase;
143 theVars->RROEA_Pool.SBPase = PS::getV9() * 1000. * 60. / SA_SBPase / mw_SBPase;
144 theVars->RROEA_Pool.PRK = PS::getV13() * 1000. * 60. / SA_PRK / mw_PRK;

146 theVars->RROEA_Pool.ATPGPP = PS::getV23() * 1000. * 60. / SA_ATPGPP / mw_ATPGPP;
147 theVars->RROEA_Pool.MDH = MDH_Vmax * 1000. * 60. / SA_MDH / mw_MDH;
148 theVars->RROEA_Pool.ThioT = 0.081;

150 theVars->RROEA_Pool.RuACTT = 0.0056;

155 double Fd = 0.081 * Coeff; // The initial concentraiton of reduced fd

157 RROEA_con->Coeff = Coeff;
158 RROEA_con->GAPDH = theVars->RROEA_Pool.GAPDH * Coeff; // The initial concentration of active GAPDH
159 RROEA_con->FBPase = theVars->RROEA_Pool.FBPase * Coeff; // The initial concentration of active FBPase
160 RROEA_con->SBPase = theVars->RROEA_Pool.SBPase * Coeff; // The initial concentration of active SBPase
161 RROEA_con->PRK = theVars->RROEA_Pool.PRK * Coeff; // The initial concentration of actove PRK
162 RROEA_con->ATPase = theVars->RROEA_Pool.ATPase * Coeff; // The initial concentration of actove ATPase
163 RROEA_con->ATPGPP = theVars->RROEA_Pool.ATPGPP * Coeff; // The initial concentration of actove ATPGPP
164 RROEA_con->MDH = 0.; // The initial concentration of actove MDH
165 RROEA_con->Thio = 0.081 * Coeff; // The initial concentration of reduced thioredoxin
166 RROEA_con->Fd = Fd; // The initial concentration of reduced ferrodoxin

181 theVars->RROEA_RC.ke2SBPase *= Coeff / 60.; // The rate constant of electron tranfer from thioredoxin to SBPase
182 theVars->RROEA_RC.ke2PRK *= Coeff / 60.; // The rate constant of electron transfer from thioredoxin to PRK, Phosphoribulase kinase
183 theVars->RROEA_RC.ke2RubACT *= Coeff / 60.; // The rate constant of electron transfer from thioredoxin to Rubisco activase
184 theVars->RROEA_RC.ke2Fd *= Coeff; // The rate constant of electron transfer to fe
185 // Jmax of 180 micro mole per meter square per second.
186 theVars->RROEA_RC.keFd2Thio *= Coeff; // The rate constant of electron transfer from fd to thio
187 theVars->RROEA_RC.keFd2Calvin *= Coeff; // The rate constant of electron transfer from fd to Calvin cycle

196 // fixed.

199 double fc16;
200 double FC;
201 if (!theVars->RROEA_EPS_com) {

203 FC = RROEA::FC;
204 fc16 = RROEA::fc16;
205 SC = RROEA::SC;

207 PS::setV3(5.04 * SC); // (Harris & Koniger, 1997) 3 GAP dehydragenase DPGA+NADPH <->GAP + OP+NADP
208 PS::setV6(1.155 * SC); // (Harris & Koniger, 1997) 6 FBPase FBP<->F6P+OP 1.155
