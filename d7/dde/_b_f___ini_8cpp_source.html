48void BF::_initCalc(Variables *theVars, BFCondition* BF_con) {
49 BF::setFI_connect(theVars->BF_FI_com);
50 BF::setPS_connect(theVars->FIBF_PSPR_com);
51 BF::setRROEA_connect(theVars->RROEA_EPS_com);
52 if (theVars->useC3) {
53 if (theVars->lightParam == 0.) {
54 const double light_scaler = theVars->alfa * (1. - theVars->fc);
55 theVars->lightParam = theVars->TestLi * 30. * light_scaler;
56 }
57 // ISPHr + cytc1 --> ISPHox + cytc1-
58 double DeltaEm = theVars->BF_RC.Em_Cytf - theVars->BF_RC.Em_IPS;
59 double DeltaG = DeltaEm * -9.649 * pow(10., 4.);
60 const double KE8 = exp(-DeltaG / BF::RT); // ISPHr + cytc1 --> ISPHox + cytc1- Unit: s-1
61
62 // cytc1- + cytc2 --> cytc1 + cytc2-
63 DeltaEm = theVars->BF_RC.Em_PG - theVars->BF_RC.Em_Cytf;
64 DeltaG = DeltaEm * -9.649 * pow(10., 4.);
65 const double KE9 = exp(-DeltaG / BF::RT); // cytc1- + cytc2 --> cytc1 + cytc2- Unit: s-1
66
67 // Assign values to the array for rate constant
68 theVars->BF_RC.K1 = theVars->EnzymeAct.at("K1"); // The rate constant for formation of ISP.QH2 complex; unit: per second
69 theVars->BF_RC.K2 = theVars->EnzymeAct.at("K2"); // The rate constant for ISP.QH2-->QH(semi) + ISPH(red) ; unit: per second
70 theVars->BF_RC.K3 = theVars->EnzymeAct.at("K3"); // The rate constant for QH. + cytbL --> Q + cytbL- + H+ Unit: s-1
71 theVars->BF_RC.K4 = theVars->EnzymeAct.at("K4"); // The rate constant for cytbL- + cytbH --> cytbL + cytbH- Unit: s-1
72 theVars->BF_RC.K5 = theVars->EnzymeAct.at("K5"); // The rate constant for CytbH- + Q --> cytbH + Q- Unit: s-1
73 theVars->BF_RC.K6 = theVars->EnzymeAct.at("K6"); // The rate constant for CytbH- + Q- --> cytbH + Q2- Unit: s-1
74 theVars->BF_RC.K7 = theVars->EnzymeAct.at("K7"); // The rate constant for Q binding to Qi site; which assumed half time as 200 us, following Croft's website Unit: s-1
75 theVars->BF_RC.K8 = theVars->EnzymeAct.at("K8"); // The rate constant for ISPH + CytC1 --> ISPH(ox) + CytC1+ Unit: s-1
76 theVars->BF_RC.K9 = theVars->EnzymeAct.at("K9"); // The rate constant for the electron transport from cytc1 to cytc2 Unit: s-1
77 theVars->BF_RC.K10 = theVars->EnzymeAct.at("K10"); // The rate constant for the electron transport from cytc2 to P700 Unit: s-1
78 theVars->BF_RC.Vmax11 = theVars->EnzymeAct.at("Vmax11") * BF::cATPsyn; // The maximum rate of ATP synthesis Unit: mmol l-1 s-1; The unit for the reactions occurrs in stroma is mmol l-1 s-1
79 theVars->BF_RC.PK = 3.6 * pow(10., -8.) * BF::PMODTEM; // The permeability constant for K Unit: cm s-1
80 theVars->BF_RC.PMg = 3.6 * pow(10., -8.) * BF::PMODTEM; // The permeability constant for Mg Unit: cm s-1
81 theVars->BF_RC.PCl = 1.8 * pow(10., -8.) * BF::PMODTEM; // The permeability constant for Cl Unit: cm s-1
82 theVars->BF_RC.Kau = theVars->EnzymeAct.at("Kau"); // The rate constant for exciton transfer from perpheral antenna to core antenna, see FI Unit: s-1
83 theVars->BF_RC.Kua = theVars->EnzymeAct.at("Kua"); // The rate constant for exciton transfer from core antenna to peripheral antenna, SEE FI Unit: s-1
84 theVars->BF_RC.Kf = theVars->EnzymeAct.at("Kf"); // The rate constant for fluorescence emission, see the note in FI Unit: s-1
85 theVars->BF_RC.Kd = theVars->EnzymeAct.at("Kd"); // The rate constant for heat dissipation; see the note for FI Unit: s-1
86 theVars->BF_RC.KE8 = KE8; // ISPHr + cytc1 --> ISPHox + cytc1- Unit: s-1
87 theVars->BF_RC.KE9 = KE9; // cytc1- + cytc2 --> cytc1 + cytc2- Unit: s-1
88 theVars->BF_RC.K15 = theVars->EnzymeAct.at("K15") * BF::CPSi; // The rate constant for primary charge separation in PSI Unit: s-1
89 theVars->BF_RC.K16 = theVars->EnzymeAct.at("K16"); // The rate constant for electron tranfer from electron acceptor of PSI to Fd Unit: s-1
90 theVars->BF_RC.V2M = theVars->EnzymeAct.at("V2M") * BF::cNADPHsyn; // The maximum rate of NADPH formation Unit: mmol l-1 s-1; The unit for the reactions occurrs in stroma is mmol l-1 s-1
91 } else {
92 // ISPHr + cytc1 --> ISPHox + cytc1-
93 double DeltaEm = theVars->BF_RC.Em_Cytf * theVars->BFRatio[22] - theVars->BF_RC.Em_IPS * theVars->BFRatio[21];
94 double DeltaG = DeltaEm * -9.649 * pow(10., 4.);
95 const double KE8 = exp(-DeltaG / BF::RT);
96
97 // cytc1- + cytc2 --> cytc1 + cytc2-
98 DeltaEm = theVars->BF_RC.Em_PG * theVars->BFRatio[25] - theVars->BF_RC.Em_Cytf * theVars->BFRatio[22];
99 DeltaG = DeltaEm * -9.649 * pow(10., 4.);
100
101 const double KE9 = exp(-DeltaG / BF::RT);
102
103 theVars->BF_RC.K1 *= theVars->BFRatio[0]; // The rate constant for formation of ISP.QH2 complex; unit: per second
104 theVars->BF_RC.K2 *= theVars->BFRatio[1]; // The rate constant for ISP.QH2-->QH(semi) + ISPH(red) ; unit: per second
105 theVars->BF_RC.K3 *= theVars->BFRatio[2]; // The rate constant for QH. + cytbL --> Q + cytbL- + H+ Unit: s-1
106 theVars->BF_RC.K4 *= theVars->BFRatio[3]; // The rate constant for cytbL- + cytbH --> cytbL + cytbH- Unit: s-1
107 theVars->BF_RC.K5 *= theVars->BFRatio[4]; // The rate constant for CytbH- + Q --> cytbH + Q- Unit: s-1
108 theVars->BF_RC.K6 *= theVars->BFRatio[5]; // The rate constant for CytbH- + Q- --> cytbH + Q2- Unit: s-1
109 theVars->BF_RC.K7 *= theVars->BFRatio[6]; // The rate constant for Q binding to Qi site; which assumed half time as 200 us, following Croft's website Unit: s-1
110 theVars->BF_RC.K8 *= theVars->BFRatio[7]; // The rate constant for ISPH + CytC1 --> ISPH(ox) + CytC1+ Unit: s-1
111 theVars->BF_RC.K9 *= theVars->BFRatio[8]; // The rate constant for the electron transport from cytc1 to cytc2 Unit: s-1
112 theVars->BF_RC.K10 *= theVars->BFRatio[9]; // The rate constant for the electron transport from cytc2 to P700 Unit: s-1
113 theVars->BF_RC.Vmax11 *= theVars->BFRatio[10]; // The maximum rate of ATP synthesis Unit: mmol l-1 s-1; The unit for the reactions occurrs in stroma is mmol l-1 s-1
114 theVars->BF_RC.Kqi *= theVars->BFRatio[11]; // The rate constant for uptake of two protons from the stroma to Q2- s-1
115 theVars->BF_RC.PK *= PMODTEM * theVars->BFRatio[12]; // The permeability constant for K Unit: cm s-1
116 theVars->BF_RC.PMg *= PMODTEM * theVars->BFRatio[13]; // The permeability constant for Mg Unit: cm s-1
117 theVars->BF_RC.PCl *= PMODTEM * theVars->BFRatio[14]; // The permeability constant for Cl Unit: cm s-1
118 theVars->BF_RC.Kau *= theVars->BFRatio[15]; // The rate constant for exciton transfer from perpheral antenna to core antenna, see FI Unit: s-1
119 theVars->BF_RC.Kua *= theVars->BFRatio[16]; // The rate constant for exciton transfer from core antenna to peripheral antenna, SEE FI Unit: s-1
120 theVars->BF_RC.Kf *= theVars->BFRatio[17]; // The rate constant for fluorescence emission, see the note in FI Unit: s-1
121 theVars->BF_RC.Kd *= theVars->BFRatio[18]; // The rate constant for heat dissipation; see the note for FI Unit: s-1
122 theVars->BF_RC.KE8 = KE8; // ISPHr + cytc1 --> ISPHox + cytc1- Unit: s-1
123 theVars->BF_RC.KE9 = KE9; // cytc1- + cytc2 --> cytc1 + cytc2- Unit: s-1
124 theVars->BF_RC.K15 *= theVars->BFRatio[19]; // The rate constant for primary charge separation in PSI Unit: s-1
125 theVars->BF_RC.K16 *= theVars->BFRatio[20]; // The rate constant for electron tranfer from electron acceptor of PSI to Fd Unit: s-1
126 theVars->BF_RC.MemCap *= theVars->BFRatio[26]; // The membrane capacity
127 theVars->BF_RC.RVA *= theVars->BFRatio[27]; // The ratio of lumen volume to thylakoid membrane area
128 theVars->BF_RC.KBs *= theVars->BFRatio[28]; // The buffer equilibrium constant in stroma
129 theVars->BF_RC.KBl *= theVars->BFRatio[29]; // The buffer equilibrium constant in lumen
130 theVars->BF_RC.KM1ATP *= theVars->BFRatio[30]; // The michaelis menton constant for ATP for ATP synthesis
131 theVars->BF_RC.KM1ADP *= theVars->BFRatio[31]; // The michaelis menton constant for ATP for ADP synthesis
132 theVars->BF_RC.KM1PI *= theVars->BFRatio[32]; // The michaelis menton constant for ATP for PI synthesis
133 theVars->BF_RC.KM2NADP *= theVars->BFRatio[33]; // The michaelis menten constant for NADP Unit: mmol l-1 s-1; The unit for the reactions occurrs in stroma is mmol l-1 s-1
134 theVars->BF_RC.KM2NADPH *= theVars->BFRatio[34]; // The michaelis menten constant for NADPH Unit: mmol l-1 s-1; The unit for the reactions occurrs in stroma is mmol l-1 s-1
135 theVars->BF_RC.V2M *= theVars->BFRatio[35]; // The maximum rate of NADPH formation Unit: mmol l-1 s-1; The unit for the reactions occurrs in stroma is mmol l-1 s-1
136 theVars->BF_RC.KE2 *= theVars->BFRatio[36]; // Equilibrium constatn
137
138 theVars->BF_Pool.kA_d *= theVars->BFRatio[37]; // The total amount of cytbH or cytbL; Unit: micromole m-2 leaf area
139 theVars->BF_Pool.kA_f *= theVars->BFRatio[38]; // The total amount of cytc; Unit: micromole m-2 leaf area
140 theVars->BF_Pool.kA_U *= theVars->BFRatio[39]; // The total concentration of K in both stroma and lumen. Unit: mmol l-1. In this model, it was assumed that the total concentration of K, and Mg and Cl as well, is constant.
141 theVars->BF_Pool.kU_A *= theVars->BFRatio[40]; // The total concentration of Mg in both stroma and lumen. Unit: mmol l-1. In this model, it was assumed that the total concentration of Mg, and K and Cl as well, is constant.
142 theVars->BF_Pool.kU_d *= theVars->BFRatio[41]; // The total concentration of Cl in both stroma and lumen. Unit: mmol l-1. In this model, it was assumed that the total concentration of Cl in both stroma and lumen is constant.
143 theVars->BF_Pool.kU_f *= theVars->BFRatio[42]; // The total concentration of Ferrodoxin
144 theVars->BF_Pool.k1 *= theVars->BFRatio[43]; // The total concentration of the primary electron acceptor of PSI; Unit: micromole m-2 leaf area
145 theVars->BF_Pool.k_r1 *= theVars->BFRatio[44]; // The total concentration of plastoquinone in thylakoid membrane. ; Unit: micromole m-2 leaf area
146 theVars->BF_Pool.kz *= theVars->BFRatio[45]; // The total concentration of buffer in stroma; unit: mmol per liter
147 theVars->BF_Pool.k12 *= theVars->BFRatio[46]; // The total concentration of buffer in lumen; unit: mmol per liter
148 theVars->BF_Pool.k23 *= theVars->BFRatio[47]; // The total number of P700; unit: micromole m-2 leaf area
149 theVars->BF_Pool.k30 *= theVars->BFRatio[48]; // The total concentration of NADPH in stroma; 1 is an guessed value;
150
151 }
