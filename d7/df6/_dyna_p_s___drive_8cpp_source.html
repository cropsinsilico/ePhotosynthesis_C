65void DynaPSDriver::setup() {
66 // This part include the function to begin the simulation.
67
68 // The time information is set in a global variable called tglobal in SYSInitial.
69 if (ParaNum <= 103) {
70 inputVars->PSRatio[ParaNum] = Ratio;
71 } else if (ParaNum <= 169) {
72 inputVars->SUCSRatio[ParaNum - 103] = Ratio;
73 } else if (ParaNum <= 217) {
74 inputVars->PRRatio[ParaNum - 169] = Ratio;
75 } else if (ParaNum <= 233) {
76 inputVars->RuACTRatio[ParaNum - 217] = Ratio;
77 } else if (ParaNum <= 256) {
78 inputVars->FIRatio[ParaNum - 233] = Ratio;
79 } else if (ParaNum <= 305) {
80 inputVars->BFRatio[ParaNum - 256] = Ratio;
81 } else if (ParaNum <= 309) {
82 inputVars->XanCycleRatio[ParaNum - 305] = Ratio;
83 }
84
85 // This part include the function to begin the simulation.
86 // The time information is set in a global variable called tglobal in SYSInitial.
87
88 SYSInitial(inputVars);
89
91 // Calculation step //
93
94 IniModelCom(inputVars); // Initialize the structure of the model, i.e. Is this model separate or combined with others.
95
96 // The combination of BF and FI model
97 inputVars->BF_FI_com = true;
98
99 // This is a variable indicating whether the PR model is actually need to be combined with PS or not. If 1 then means combined; 0 means not.
100 inputVars->PR_PS_com = true;
101
102 // 1 means that the overall EPS model is used. 0 means partial model of FIBF is used.
103 inputVars->FIBF_PSPR_com = true;
104
105 // A global variable to indicate whether the RuACT is run by itself or combined with others.
106 // langmm: This is true in the MATLAB code, but was false in the
107 // original translation to C++
108 inputVars->RuACT_EPS_com = true; // Since this is run within this program, it is combinbed, therefore, it is assigned value true, otherwise, assign value false.
109
110 // This is the connection between Redox and RA.
111 inputVars->RedoxReg_RA_com = false; // This means that the connection is not provided there.
112
113 inputVars->XanCycle_BF_com = true;
114
115 inputVars->EPS_SUCS_com = true;
116
117 // This is a variable indicating whether the PSPR model is actually need to be combined with SUCS or not. If 1 then means combined; 0 means not.
118 inputVars->PSPR_SUCS_com = true;
119
120 inputVars->SUCS_Param = zeros(2);
121
122 // Next is to initialize the vector.
123
124 DynaPSCondition* DynaPS_con = DynaPS_Ini();
125
126 const double va1 = 0;
127 // The ratio of the PSI unit to the PSII unit
128 inputVars->BF_Param[0] = va1;
129 inputVars->BF_Param[1] = inputVars->PS12ratio;
130
131 inputVars->FI_Param[0] = va1;
132 inputVars->FI_Param[1] = inputVars->PS12ratio;
133
134 inputVars->PS_PR_Param = 0.;
135
136 inputVars->RuACT_Param[0] = va1;
137 inputVars->RuACT_Param[1] = inputVars->PS12ratio;
138
139 inputVars->XanCycle_Param[0] = va1;
140 inputVars->XanCycle_Param[1] = inputVars->PS12ratio;
141
142 constraints = DynaPS_con->toArray();
143 delete DynaPS_con;
144}

146void DynaPSDriver::getResults() {
147
148 DynaPSCondition* dyna_int_con = new DynaPSCondition(intermediateRes);
149
150 arr temp = DynaPS::MB(time, dyna_int_con, inputVars);
151
152 double PSIIabs = inputVars->FI_Vel.vP680_d;
153 double PSIabs = inputVars->BF_Vel.Vbf11;
154 double CarbonRate = inputVars->RuACT_Vel.v6_1 * inputVars->AVR;
155 double VPR = inputVars->RuACT_Vel.v6_2 * inputVars->AVR;
156 double Vpgasink = inputVars->SUCS_Vel.vpga_use * inputVars->AVR;
157 double VStarch = (inputVars->PS_Vel.v23 - inputVars->PS_Vel.v25) * inputVars->AVR;
158 double Vsucrose = inputVars->SUCS_Vel.vdhap_in * inputVars->AVR;
159 const double CO2AR = TargetFunVal(inputVars);
160
161 results = zeros(8);
162
163 results[0] = PSIIabs;
164 results[1] = PSIabs;
165 //Resulta(3)=PSIabs2(row);
166 results[2] = CarbonRate;
167 results[3] = VPR;
168 results[4] = Vpgasink;
169 results[5] = Vsucrose;
170 results[6] = VStarch;
171 results[7] = CO2AR;
172
173 if(inputVars->record) {
174 makeFluxTR(inputVars);
175 }
176 // This is to set the regualtions to be as beginning.
177 //theVars->ATPActive = 0;
178 inputVars->BF_FI_com = false;
179 inputVars->PR_PS_com = false;
180 inputVars->FIBF_PSPR_com = false;
181 inputVars->RuACT_EPS_com = false;
182 inputVars->RedoxReg_RA_com = false;
183 inputVars->XanCycle_BF_com = false;
184
185 delete dyna_int_con;
186 IniModelCom(inputVars);
187 //save FDC2
188}
