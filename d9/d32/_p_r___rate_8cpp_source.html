37void PR::_Rate(const double t, const PRCondition* const PR_con, Variables *theVars) {
38 // To set global information for different reactions
39
40 // Reaction: 111: RUBP+O2<-->PGlycolate + PGA
41
42 // Reaction: 112: PGlycolate-->Pi+Glycolate;
43
44 // Reaction 113 : Gla+ATP<-->ADP + PGA
45
46 // Reactoin 121; Glycolate +O2<-->H2O2+Glyoxylate
47
48 // Reaction 122 : Glyoxylate + Serine<--> Hydoxypyruvate + Glycine;
49 // Reaction 123: HydroxylPyruvate + NAD <--> NADH + Glycerate
50
51 // Reaction 124: Glyoxylate + Glu <--> KG + Glycine;
52
53 // Reaction 131: LS2+Glycine <--> CO2+ AMDHL
54
55 // Calculate the reactino inside chloroplast;
56 double RuBP;
57 if (PR::PS_RuBP) {
58 RuBP = PR_con->parent->PS_con->RuBP;
59 } else {
60 RuBP = PR_con->RuBP;
61 }
62
63 if (theVars->useC3) {
64
65 const double PrV111 = PS::getPsV1() * 0.24;
66
67 if (theVars->PR_PS_com) {
68 PR::KC= PS::getKM11();
69 PR::KO= PS::getKM12();
70 }
71
72 if (theVars->RUBISCOMETHOD == 2) { // Using michelies and enzyme information
73 double PrV111t;
74 if (theVars->PR_PS_com) { // FOr the combined PS-PR model
75 PrV111t = PrV111*RuBP / (RuBP + PR::KR * PS::getV1Reg());
76 } else { // For the PR model
77 PrV111t = PrV111*RuBP / (RuBP + PR::KR);
78 }
79 theVars->PR_Vel.v111 = PrV111t * theVars->O2_cond / (theVars->O2_cond + PR::KO *
80 (1. + theVars->CO2_cond / PR::KC));
81
82#ifndef MAKE_EQUIVALENT_TO_MATLAB
83 // This is not present for RUBISCOMETHOD == 2 in the Matlab code
84 if (RuBP < PR::RUBISCOTOTAL)
85 theVars->PR_Vel.v111 = theVars->PR_Vel.v111 * RuBP / PR::RUBISCOTOTAL;
86#endif // MAKE_EQUIVALENT_TO_MATLAB
87
88 } else if (theVars->RUBISCOMETHOD == 1) {
89 theVars->PR_Vel.v111 = PrV111 * theVars->O2_cond / (theVars->O2_cond + PR::KO *
90 (1. + theVars->CO2_cond / PR::KC));
91 if (RuBP < PR::RUBISCOTOTAL)
92 theVars->PR_Vel.v111 = theVars->PR_Vel.v111 * RuBP / PR::RUBISCOTOTAL;
93
94 }
95
96 theVars->PR_Vel.v112 = PR::PrV112 * PR_con->PGCA / (PR_con->PGCA + PR::KM112 *
97 (1. + PR_con->GCA / PR::KI1122) *
98 (1. + theVars->Pi / PR::KI1121));
99
100 if (theVars->PR_PS_com) { // For the combined PS-PR MODEL
101 theVars->PR_Vel.v113 = PR::PrV113 * (PR_con->parent->PS_con->ATP * PR_con->GCEA -
102 theVars->ADP * PR_con->parent->PS_con->PGA / PR::KE113) /
103 ((PR_con->parent->PS_con->ATP + PR::KM1131 *
104 (1. + PR_con->parent->PS_con->PGA / PR::KI113)) *
105 (PR_con->GCEA + PR::KM1132));
106 } else {
107 theVars->PR_Vel.v113 = PR::PrV113 * (PR_con->parent->PS_con->ATP * PR_con->GCEA -
108 theVars->ADP * PR::PGA / PR::KE113) /
109 ((PR_con->parent->PS_con->ATP + PR::KM1131 *
110 (1. + 2.5 / PR::KI113)) * (PR_con->GCEA + PR::KM1132));
111 }
112
113 theVars->PR_Vel.v121 = PR::PrV121 * PR_con->GCAc / (PR_con->GCAc + PR::KM121);
114
115 theVars->PR_Vel.v122 = PR::PrV122 * (PR_con->GOAc * PR_con->SERc - PR_con->HPRc *
116 PR_con->GLYc / PR::KE122) / ((PR_con->GOAc + PR::KM1221) *
117 (PR_con->SERc + PR::KM1222 *
118 (1. + PR_con->GLYc / PR::KI1221)));
119 theVars->PR_Vel.v123 = PR::PrV123 * (PR_con->HPRc * PR::NADHc - PR_con->GCEA * PR::NADc / PR::KE123) /
120 (PR_con->HPRc + PR::KM123 * (1. + PR_con->HPRc / PR::KI123));
121 theVars->PR_Vel.v124 = PR::PrV124 * (PR_con->GOAc * PR::GLUc - PR::KGc * PR_con->GLYc / PR::KE124) /
122 ((PR_con->GOAc + PR::KM1241) * (PR::GLUc + PR::KM1242 *
123 (1. + PR_con->GLYc / PR::KI124)));
124
125 theVars->PR_Vel.v131 = PR::PrV131 * PR_con->GLYc/(PR_con->GLYc + PR::KM1311 *
126 (1. + PR_con->SERc / PR::KI1311));
127 theVars->PR_Vel.v2out = PR::V2T * (PR_con->GCA/(PR_con->GCA + PR::KM1012 *
128 (1. + PR_con->GCEA / PR::KI1012)) -
129 PR_con->GCAc /(PR_con->GCAc + PR::KM1012 *
130 (1. + PR_con->GCEAc / PR::KI1012))); // Competive inhibition
131 theVars->PR_Vel.v1in = PR::V1T *(PR_con->GCEAc / (PR_con->GCEAc + PR::KM1011 *
132 (1. + PR_con->GCAc / PR::KI1011)) -
133 PR_con->GCEA /(PR_con->GCEA + PR::KM1011 *
134 (1. + PR_con->GCA / PR::KI1011))); // Competive inhibition
135
136 } else {
137 theVars->Pi = theVars->PR_Param[1]; // Value from spinach. Assume constant currently.
138
139 // Information from PS cycle is transfered back to the PR cycle for calcualtion.
140
141 double ATP;
142 double ADP;
143 if (theVars->PR_PS_com) {
144 ADP = theVars->ADP;
145 ATP = PR_con->parent->PS_con->ATP;
146 } else {
147 ADP = PR::PR_ADP;
148 ATP = PR::PR_ATP;
149 }
150
151 double PrV111t;
152 if (theVars->RUBISCOMETHOD == 2) {
153 if (theVars->PR_PS_com) {
154 PrV111t = PR::V111 * RuBP / (RuBP + PR::KR * PS::getV1Reg());
155 } else {
156 PrV111t = PR::V111 * RuBP / (RuBP + PR::KR);
157 }
158 theVars->PR_Vel.v111 = PrV111t * theVars->O2_cond / (theVars->O2_cond + PR::KO *
159 (1. + theVars->CO2_cond / PR::KC));
160
161 if (RuBP < PS::getPsV1() / 2.5)
162 theVars->PR_Vel.v111 = theVars->PR_Vel.v111 * (2.5 * RuBP / PrV111t);
163
164
165 } else if (theVars->RUBISCOMETHOD == 1){
166 theVars->PR_Vel.v111 = PR::V111 * theVars->O2_cond / (theVars->O2_cond + PR::KO *
167 (1. + theVars->CO2_cond / PR::KC));
168 if (RuBP < PR::RUBISCOTOTAL)
169 theVars->PR_Vel.v111 = theVars->PR_Vel.v111 * RuBP / PR::RUBISCOTOTAL;
170
171 } else {
172 throw std::invalid_argument("Invalid value of RUBISCOMETHOD used");
173 }
174
175 //double v113;
176 if (theVars->PR_PS_com) {
177 theVars->PR_Vel.v113 = PR::V113 * (ATP * PR_con->GCEA - ADP * PR_con->parent->PS_con->PGA / PR::KE113) /
178 ((ATP + PR::KM1131 * (1. + PR_con->parent->PS_con->PGA / PR::KI113)) *
179 (PR_con->GCEA + PR::KM1132));// This is the old version.
180
181 } else {
182 theVars->PR_Vel.v113 = PR::V113 * (ATP * PR_con->GCEA - ADP * PR::PGA / PR::KE113) /
183 ((ATP + PR::KM1131 * (1. + 2.5 / PR::KI113)) *
184 (PR_con->GCEA + PR::KM1132));// This is the old version.
185 }
186
187
188 // The reaction: PR_con->GOAc + Serine --> HPR + Gly
189
190 const double PrKM1232 = 0.5;
191
192 // The following is used to take the information back to the PRmb routine.
193 theVars->PR_Vel.v112 = PR::V112 * PR_con->PGCA / (PR_con->PGCA + PR::KM112 *
194 (1. + PR_con->GCA / PR::KI1122) *
195 (1. + theVars->Pi / PR::KI1121));
196 theVars->PR_Vel.v121 = PR::V121 * PR_con->GCAc / (PR_con->GCAc + PR::KM121);
197 theVars->PR_Vel.v122 = PR::V122 * (PR_con->GOAc * PR_con->SERc - PR_con->HPRc * PR_con->GLYc /
198 PR::KE122) / ((PR_con->GOAc + PR::KM1221) *
199 (PR_con->SERc + PR::KM1222 *
200 (1. + PR_con->GLYc / PR::KI1221)));
201 theVars->PR_Vel.v123 = PR::V123 * (PR_con->HPRc * PR::NADHc - PR_con->GCEAc * PR::NADc / PR::KE123) /
202 ((PR_con->HPRc + PR::KM123 * (1. + PR_con->HPRc / PR::KI123)) *
203 (PR::NADHc + PrKM1232));
204 theVars->PR_Vel.v124 = PR::V124 * (PR_con->GOAc * PR::GLUc - PR::KGc * PR_con->GLYc / PR::KE124) /
205 ((PR_con->GOAc + PR::KM1241) * (PR::GLUc + PR::KM1242 *
206 (1. + PR_con->GLYc / PR::KI1221)));
207 theVars->PR_Vel.v131 = PR::V131 * PR_con->GLYc / (PR_con->GLYc + PR::KM1311 *
208 (1. + PR_con->SERc / PR::KI1311));
209 theVars->PR_Vel.v1in = PR::V1T * (PR_con->GCEAc / (PR_con->GCEAc + PR::KM1011 *
210 (1. + PR_con->GCAc / PR::KI1011)) -
211 PR_con->GCEA / (PR_con->GCEA + PR::KM1011 *
212 (1. + PR_con->GCA / PR::KI1011)));// Competive inhibition
213 theVars->PR_Vel.v2out = PR::V2T * (PR_con->GCA / (PR_con->GCA + PR::KM1012 *
214 (1. + PR_con->GCEA / PR::KI1012)) -
215 PR_con->GCAc / (PR_con->GCAc + PR::KM1012 *
216 (1. + PR_con->GCEAc / PR::KI1012)));// Competive inhibition
217 }
218
219#ifdef INCDEBUG
220 DEBUG_INTERNAL(theVars->PR_Vel)
221#endif
222 if (theVars->record) {
223 if (t > PR::TIME) {
224 PR::N++;
225 PR::TIME = t;
226 }
227
228 theVars->PR_VEL.insert(PR::N - 1, t, theVars->PR_Vel);
229
230 // here is some parameters we need to output
231
232 theVars->PR2OUT.GCEA = PR_con->GCEA;
233 theVars->PR2OUT.GCA = PR_con->GCA;
234 theVars->PR2OUT.PGCA = PR_con->PGCA;
235 theVars->PR2OUT.GCAc = PR_con->GCAc;
236 theVars->PR2OUT.GOAc = PR_con->GOAc;
237 theVars->PR2OUT.SERc = PR_con->SERc;
238 theVars->PR2OUT.GLYc = PR_con->GLYc;
239 theVars->PR2OUT.HPRc = PR_con->HPRc;
240 theVars->PR2OUT.GCEAc = PR_con->GCEAc;
241 if (!PR::PS_RuBP)
242 theVars->PR2OUT.RuBP = RuBP;
243 theVars->PR2OUT._v131 = theVars->PR_Vel.v131;
244 }
