38void FI::_Rate(const double t, const FICondition* const FI_Con, Variables *theVars) {
39
41 // Step 1 Get the rate constant and the initial concentrations //
43
44 // The rate constant used in the model
45 double PQn;
46 double PQ;
47 if (FICondition::BF_connect) {
48 PQn = FI_Con->parent->BF_con->QH2;
49 PQ = FIBF::getFIBF2FI_PQ();
50 } else {
51 PQn = FI_Con->PQn;
52 PQ = theVars->FI_Pool.PQT - PQn - FIBF::getFIBF2FI_PQa();
53 }
54
55 const double P680Pheo = P680PheoT - FI_Con->P680pPheo - FI_Con->P680Pheon - FI_Con->P680pPheon -
56 FI_Con->P680ePheo; //QF add '- P680ePheo'
57
58 //const double It = theVars->FI_Param[0]; // It The total incident light intensity
59
60 // rate: FI_Con.U -> U*
61 theVars->FI_Vel.Ic = theVars->FI_Param[0] * FIBF::getChlT() / (FIBF::getChlT2() +
62 FIBF::getChlPSI()); // Ic The incident light on the core antenna; theVars->ChlT is defined in upper lines as the total amount of Chl in one U.
63 // rate: FI_Con.A -> A*
64 theVars->FI_Vel.Ia = theVars->FI_Param[0] * (FIBF::getChlT2() - FIBF::getChlT()) /
65 (FIBF::getChlT2() + FIBF::getChlPSI()); // Ia The incident light on the peripheral antenna
66
68 // Calculate the rate of different reactions //
70
71 const double q = (FI_Con->QAQB + FI_Con->QAQBn + FI_Con->QAQB2n) / (FI_Con->QAQB + FI_Con->QAQBn +
72 FI_Con->QAQB2n + FI_Con->QAnQB +
73 FI_Con->QAnQBn + FI_Con->QAnQB2n);
74 theVars->FI_Vel.vA_d = FI_Con->A * theVars->FI_RC.kA_d; // vA_d The rate of heat dissipation from peripheral antenna
75 theVars->FI_Vel.vA_f = FI_Con->A * theVars->FI_RC.kA_f; // vA_f The rate of fluorescence emission from peripheral antenna
76 theVars->FI_Vel.vA_U = FI_Con->A * theVars->FI_RC.kA_U; // vA_U The rate of exciton transfer from peripheral antenna to core antenna in open reaction center
77 theVars->FI_Vel.vU_A = FI_Con->U * theVars->FI_RC.kU_A; // vU_A The rate of exciton transfer from core antenna to perpheral antenna in open center
78 theVars->FI_Vel.vU_f = FI_Con->U * theVars->FI_RC.kU_f; // vU_f The rate of fluorescence emission from core antenna
79 theVars->FI_Vel.vU_d = FI_Con->U * theVars->FI_RC.kU_d * (1. - q); // vU_d The rate of heat dissipation from core antenna
80
81 const double P = 1.;
82
83 // energy flow:
84 theVars->FI_Vel.vU_P680 = theVars->FI_Vel.Ic + theVars->FI_Vel.vA_U - theVars->FI_Vel.vU_A -
85 theVars->FI_Vel.vU_f - theVars->FI_Vel.vU_d;//QF add , total energy coming to P680 and == the rate of P680 -> P680*, except f and d, energy is transport to P680 reaction center
86 theVars->FI_Vel.vP680_d = FI_Con->P680ePheo * theVars->FI_RC.kU_d * (1. - q);
87 theVars->FI_Vel.v1 = FI_Con->P680ePheo * theVars->FI_RC.k1 * q + FI_Con->P680ePheo * P *
88 (1. - q) * theVars->FI_RC.k1 / 6.2 + FI_Con->P680ePheo * (1. - P) *
89 (1. - q) * theVars->FI_RC.k1; // v1 The rate of primary charge separation
90
91 theVars->FI_Vel.v_r1 = FI_Con->P680pPheon * theVars->FI_RC.k_r1 * q + FI_Con->P680pPheon *
92 (1. - q) * theVars->FI_RC.k_r1 * 3.; // v_r1 The rate of charge recombination
93 theVars->FI_Vel.vP680_f = theVars->FI_Vel.vU_P680 - (theVars->FI_Vel.v1 - theVars->FI_Vel.v_r1) -
94 theVars->FI_Vel.vP680_d;
95
96 double Coeff = FI_Con->P680pPheon / P680PheoT;
97 theVars->FI_Vel.v1z_1 = FI_Con->S1T * theVars->FI_RC.kz * Coeff; // v1z_1 The rate of oxidation of FI_Con.S1T by P680pPheon
98 theVars->FI_Vel.v2z_1 = FI_Con->S2T * theVars->FI_RC.kz * Coeff; // v2z_1 The rate of oxidation of FI_Con.S2T by P680pPheon
99 theVars->FI_Vel.v3z_1 = FI_Con->S3T * theVars->FI_RC.kz * Coeff; // v3z_1 The rate of oxidation of FI_Con.S3T by P680pPheon
100 theVars->FI_Vel.v0z_1 = FI_Con->S0T * theVars->FI_RC.kz * Coeff; // v0z_1 The rate of oxidation of FI_Con.S0T by P680pPheon
101
102 theVars->FI_Vel.vz_1 = theVars->FI_Vel.v1z_1 + theVars->FI_Vel.v2z_1 + theVars->FI_Vel.v3z_1 +
103 theVars->FI_Vel.v0z_1; // vz_1 The rate of FI_Con.P680pPheon reduction
104
105 Coeff = FI_Con->P680pPheo / P680PheoT;
106
107 theVars->FI_Vel.v1z_2 = FI_Con->S1T * theVars->FI_RC.kz * Coeff; // v1z_2 The rate of oxidation of FI_Con.S1T by P680pPheo
108 theVars->FI_Vel.v2z_2 = FI_Con->S2T * theVars->FI_RC.kz * Coeff; // v2z_2 The rate of oxidation of FI_Con.S2T by P680pPheo
109 theVars->FI_Vel.v3z_2 = FI_Con->S3T * theVars->FI_RC.kz * Coeff; // v3z_2 The rate of oxidation of FI_Con.S3T by P680pPheo
110 theVars->FI_Vel.v0z_2 = FI_Con->S0T * theVars->FI_RC.kz * Coeff; // v0z_2 The rate of oxidation of FI_Con.S0T by P680pPheo
111 theVars->FI_Vel.vz_2 = theVars->FI_Vel.v1z_2 + theVars->FI_Vel.v2z_2 + theVars->FI_Vel.v3z_2 +
112 theVars->FI_Vel.v0z_2; // vz_2 The rate of FI_Con.P680pPheo reduction
113
114 theVars->FI_Vel.v1z = theVars->FI_Vel.v1z_1 + theVars->FI_Vel.v1z_2;
115 theVars->FI_Vel.v2z = theVars->FI_Vel.v2z_1 + theVars->FI_Vel.v2z_2;
116 theVars->FI_Vel.v3z = theVars->FI_Vel.v3z_1 + theVars->FI_Vel.v3z_2;
117 theVars->FI_Vel.v0z = theVars->FI_Vel.v0z_1 + theVars->FI_Vel.v0z_2;
118
119 theVars->FI_Vel.vAB1 = FI_Con->QAnQB * theVars->FI_RC.kAB1; // vAB1 The rate of electron transfer from QA- to QB
120 theVars->FI_Vel.vBA1 = FI_Con->QAQBn * theVars->FI_RC.kBA1; // vBA1 The rate of electron transfer from QB- to QA
121 theVars->FI_Vel.vAB2 = FI_Con->QAnQBn * theVars->FI_RC.kAB2; // vAB2 The rate of electron transfer from QA- to QB-
122 theVars->FI_Vel.vBA2 = FI_Con->QAQB2n * theVars->FI_RC.kBA2; // vBA2 The rate of electron transfer from QB2- TO QA
123 theVars->FI_Vel.v3 = FI_Con->QAQB2n * PQ * theVars->FI_RC.k3 / theVars->FI_Pool.PQT; // v3 The rate of exchange of QAQBH2 with PQ
124 theVars->FI_Vel.v_r3 = FI_Con->QAQB * PQn * theVars->FI_RC.k_r3 / theVars->FI_Pool.PQT; // v_r3 The rate of exchange of FI_Con.QAQB with PQH2
125
126 theVars->FI_Vel.v3_n = FI_Con->QAnQB2n * PQ * theVars->FI_RC.k3 / theVars->FI_Pool.PQT; // v3_n The rate of exchange of QAnQBH2 with PQ
127 theVars->FI_Vel.v_r3_n = FI_Con->QAnQB * PQn * theVars->FI_RC.k_r3 / theVars->FI_Pool.PQT; // v_r3_n The rate of exchange of FI_Con.QAnQB with PQH2
128
129 theVars->FI_Vel.v_pq_ox = PQn * theVars->FI_RC.k_pq_oxy; // v_pq_ox The rate of PQH2 oxidation
130
131 theVars->FI_Vel.v2_1 = FI_Con->P680pPheon * theVars->FI_RC.k2 * q; // v2_1 The rate of FI_Con.P680pPheon oxidation
132 theVars->FI_Vel.v2_2 = FI_Con->P680Pheon * theVars->FI_RC.k2 * q; // v2_1 The rate of FI_Con.P680pPheon oxidation
133 const double a = FI_Con->QAQB / (FI_Con->QAQB + FI_Con->QAQBn + FI_Con->QAQB2n);
134 const double b = FI_Con->QAQBn / (FI_Con->QAQB + FI_Con->QAQBn + FI_Con->QAQB2n);
135 const double c = FI_Con->QAQB2n / (FI_Con->QAQB + FI_Con->QAQBn + FI_Con->QAQB2n);
136
137 theVars->FI_Vel.v2_00_1 = theVars->FI_Vel.v2_1 * a; // v2_00_1 The rate of reduction of FI_Con.QAQB by P680pPheon
138 theVars->FI_Vel.v2_01_1 = theVars->FI_Vel.v2_1 * b; // v2_01_1 The rate of reduction of FI_Con.QAQBn by P680pPheon
139 theVars->FI_Vel.v2_02_1 = theVars->FI_Vel.v2_1 * c; // v2_02_1 The rate of reduction of FI_Con.QAQB2n by P680pPheon
140
141 theVars->FI_Vel.v2_00_2 = theVars->FI_Vel.v2_2 * a; // v2_00_2 The rate of reduction of FI_Con.QAQB by P680Pheon
142 theVars->FI_Vel.v2_01_2 = theVars->FI_Vel.v2_2 * b; // v2_01_2 The rate of reduction of FI_Con.QAQBn by P680Pheon
143 theVars->FI_Vel.v2_02_2 = theVars->FI_Vel.v2_2 * c; // v2_02_2 The rate of reduction of FI_Con.QAQB2n by P680Pheon
144
145 const double Coeff1 = FI_Con->P680pPheo / P680PheoT;
146 theVars->FI_Vel.vr2_00_1 = FI_Con->QAnQB * theVars->FI_RC.k2 / KE * Coeff1; // vr2_00_1 The reverse reaction of The rate of reduction of FI_Con.QAQB by P680pPheon
147 theVars->FI_Vel.vr2_01_1 = FI_Con->QAnQBn * theVars->FI_RC.k2 / KE * Coeff1; // vr2_01_1 The reverse reaction of The rate of reduction of FI_Con.QAQBn by P680pPheon
148 theVars->FI_Vel.vr2_02_1 = FI_Con->QAnQB2n * theVars->FI_RC.k2 / KE * Coeff1; // vr2_02_1 The reverse reaction of The rate of reduction of FI_Con.QAQB2n by P680pPheon
149 theVars->FI_Vel.vr2_1 = theVars->FI_Vel.vr2_00_1 + theVars->FI_Vel.vr2_01_1 + theVars->FI_Vel.vr2_02_1;
150
151 const double Coeff2 = P680Pheo / P680PheoT;// Coeff2
152 theVars->FI_Vel.vr2_00_2 = FI_Con->QAnQB * theVars->FI_RC.k2 / KE * Coeff2;// vr2_00_2 The reverse reaction of The rate of reduction of FI_Con.QAQB by P680Pheon
153 theVars->FI_Vel.vr2_01_2 = FI_Con->QAnQBn * theVars->FI_RC.k2 / KE * Coeff2;// vr2_01_2 The reverse reaction of The rate of reduction of FI_Con.QAQBn by P680Pheon
154 theVars->FI_Vel.vr2_02_2 = FI_Con->QAnQB2n * theVars->FI_RC.k2 / KE * Coeff2;// vr2_02_2 The reverse reaction of The rate of reduction of FI_Con.QAQB2n by P680Pheon
155 theVars->FI_Vel.vr2_2 = theVars->FI_Vel.vr2_00_2 + theVars->FI_Vel.vr2_01_2 + theVars->FI_Vel.vr2_02_2;// vr2_2
156
157 theVars->FI_Vel.vP680qU = pow(10., 9.) * FI_Con->U * (FI_Con->P680pPheo + FI_Con->P680pPheon) +
158 FI_Con->U * 0.15 * (theVars->FI_RC.kU_f + theVars->FI_RC.kU_d) * PQ /
159 theVars->FI_Pool.PQT;
160 theVars->FI_Vel.vP680qA = pow(10., 9.) * FI_Con->A * (FI_Con->P680pPheo + FI_Con->P680pPheon) +
161 FI_Con->A * 0.15 * (theVars->FI_RC.kA_f + theVars->FI_RC.kA_d) * PQ /
162 theVars->FI_Pool.PQT;
163
166 //const double f = vA_f + vU_f;
168
170 // Assign table
172
173 theVars->FI_Vel.vS1_S2 = FI_Con->S1Tp * theVars->FI_RC.k12; // vS1_S2 The rate of transition from S1 to S2
174 theVars->FI_Vel.vS2_S3 = FI_Con->S2Tp * theVars->FI_RC.k23; // vS2_S3 The rate of transition from S2 to S3
175 theVars->FI_Vel.vS3_S0 = FI_Con->S3Tp * theVars->FI_RC.k30; // vS3_S0 The rate of transition from S3 to S0
176 theVars->FI_Vel.vS0_S1 = FI_Con->S0Tp * theVars->FI_RC.k01; // vS0_S1 The rate of transition from S0 to S1
177
178#ifdef INCDEBUG
179 DEBUG_INTERNAL(theVars->FI_Vel)
180#endif
181 if (theVars->record) {
183 // Part V Output of Velocity for plot //
185
186 if (t > FI::TIME) {
187 FI::N++;
188 FI::TIME = t;
189 }
190
191 theVars->FI_VEL.insert(FI::N - 1, t, theVars->FI_Vel);
192 }
