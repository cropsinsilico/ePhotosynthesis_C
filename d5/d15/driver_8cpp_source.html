67arr Driver::run() {
68 if (origVars != nullptr) {
69 delete origVars;
70 origVars = nullptr;
71 }
72 origVars = new Variables(inputVars);
73 uint count = 0;
74
75 while (count < 10){
76 maxStep = 20. * step;
77
78 setup();
79
80 sunindextype N = static_cast<long>(constraints.size());
81 N_Vector y;
82#ifdef SUNDIALS_CONTEXT_REQUIRED
83 y = N_VNew_Serial(N, context());
84#else // SUNDIALS_CONTEXT_REQUIRED
85 y = N_VNew_Serial(N);
86#endif // SUNDIALS_CONTEXT_REQUIRED
87 sunrealtype* y_ptr = N_VGetArrayPointer(y);
88
89 for (std::size_t i = 0; i < constraints.size(); i++)
90 y_ptr[i] = constraints[i];
91 realtype t0 = start;
92
93 CVodeMem *cmem = nullptr;
94 try {
95 cmem = &CVodeMem::create();
96 cmem->cvode_mem_init(this, t0, y);
97 } catch(...) {
98 std::exception_ptr eptr = std::current_exception();
99 N_VDestroy(y);
100 std::rethrow_exception(eptr);
101 }
102
103 data->drv = this;
104
105#ifdef SUNDIALS_CONTEXT_REQUIRED
106 SUNMatrix A = SUNDenseMatrix(N, N, context());
107 SUNNonlinearSolver NLS = SUNNonlinSol_Newton(y, context());
108 SUNLinearSolver LS = SUNLinSol_Dense(y, A, context());
109#else // SUNDIALS_CONTEXT_REQUIRED
110 SUNMatrix A = SUNDenseMatrix(N, N);
111 SUNNonlinearSolver NLS = SUNNonlinSol_Newton(y);
112 SUNLinearSolver LS = SUNLinSol_Dense(y, A);
113#endif // SUNDIALS_CONTEXT_REQUIRED
114
115 try {
116 if (CVodeSetNonlinearSolver(cvode_mem, NLS) != CV_SUCCESS) {
117 std::cout << "CVodeSetNonlinearSolver failed" << std::endl;
118 throw std::runtime_error("CVodeSetNonlinearSolver failed");
119 }
120 if (CVodeSetLinearSolver(cvode_mem, LS, A) != CV_SUCCESS) {
121 std::cout << "CVodeSetLinearSolver failed" << std::endl;
122 throw std::runtime_error("CVodeSetLinearSolver failed");
123 }
124 } catch(...) {
125 std::exception_ptr eptr = std::current_exception();
126 SUNNonlinSolFree(NLS);
127 SUNLinSolFree(LS);
128 SUNMatDestroy(A);
129 N_VDestroy(y);
130 std::rethrow_exception(eptr);
131 }
132
133 realtype t = 0;
134 bool runOK = true;
135 realtype tout = start + step;
136 while (t <= endtime) {
137 if (CVode(cvode_mem, tout, y, &t, CV_NORMAL) != CV_SUCCESS) {
138 std::cout << "CVode failed at t=" << tout << " " << t << std::endl;
139 runOK = false;
140 break;
141 }
142 tout += step;
143 }
144 if (runOK) {
145 intermediateRes = N_VGetArrayPointer(y);
146 time = t;
147 _lastStep = true;
148 _dumpStep = true;
149 getOutputVars(inputVars); // Before reset called from getResults
150 getResults();
151 }
152
153 SUNNonlinSolFree(NLS);
154 SUNLinSolFree(LS);
155 SUNMatDestroy(A);
156 N_VDestroy(y);
157 if (runOK)
158 return results;
159
160 inputVars = origVars;
161
162 count++;
163 step = initialStep / (count + 1);
164 std::cout << "Retrying with smaller step size: " << step << std::endl;
165 }
166 throw std::runtime_error("No valid solution found");
167}

178void Driver::outputParam(const OutputFreq& frequency,
179 const std::string& fname_base,
180 const std::vector<std::string>& vars) {
181 std::string fname_base_cpy = fname_base;
182 if (fname_base_cpy.empty()) {
183 fname_base_cpy = "NOMODULE_";
184 }
185 std::string finit, flast, fstep;
186 if (frequency >= OUTPUT_FREQ_FIRST)
187 finit = fname_base_cpy + "init.txt";
188 if (frequency >= OUTPUT_FREQ_FIRST_AND_LAST)
189 flast = fname_base_cpy + "last.txt";
190 if (frequency >= OUTPUT_FREQ_STEPS)
191 fstep = fname_base_cpy + "step";
192 outputParam(finit, flast, fstep, vars);
193}

225double Driver::getVar(const Variables* inputVars, const std::string& k) {
226 if (k == "Light intensity")
227 return inputVars->TestLi;
228 else if (k == "Vc")
229 return inputVars->RuACT_Vel.v6_1 * inputVars->AVR;
230 else if (k == "Vo")
231 return inputVars->RuACT_Vel.v6_2 * inputVars->AVR;
232 else if (k == "VPGA")
233 return inputVars->SUCS_Vel.vpga_use * inputVars->AVR;
234 else if (k == "Vstarch")
235 return (inputVars->PS_Vel.v23 - inputVars->PS_Vel.v25) * inputVars->AVR;
236 else if (k == "Vsucrose")
237 return inputVars->SUCS_Vel.vdhap_in * inputVars->AVR;
238 else if (k == "VT3P")
239 return (inputVars->PS_Vel.v31 + inputVars->PS_Vel.v33) * inputVars->AVR;
240 else if (k == "Vt_glycerate")
241 return inputVars->PR_Vel.v1in * inputVars->AVR;
242 else if (k == "Vt_glycolate")
243 return inputVars->PR_Vel.v2out * inputVars->AVR;
244 else if (k == "PSIIabs")
245 return inputVars->FI_Vel.vP680_d;
246 else if (k == "PSIabs")
247 return inputVars->BF_Vel.Vbf11;
248 else if (k == "CO2AR")
249 return TargetFunVal(inputVars);
250 else
251 return inputVars->getVar(k);
252}

273 {"BF::POOL::kz", "BF::POOL::BFTs"},

275 {"BF::POOL::k23", "BF::POOL::P700T"},
276 {"BF::POOL::k30", "BF::POOL::NADPHT"}

278 static std::vector<std::string> skip_keys = {

280 "ALL::VARS::GLight",
281 // Not named variables in Matlab
282 "BF::RC::Em_IPS",
283 "BF::RC::Em_Cytf",
284 "BF::RC::Em_PG",
285 "PS::MOD::KE1Ratio",
286 "PS::MOD::KE2Ratio",
287 "SUCS::MOD::KE5Ratio",
288 };
289 // Only used if useC3 false which is incompatible with C3 matlab
290 if (theVars->useC3) {
291 skip_keys.push_back("PS::MOD::KE57");
292 skip_keys.push_back("PS::COND::ADPG");
293 skip_keys.push_back("SUCS::MOD::KI583");

295 skip_keys.push_back("SUCS::MOD::SC1");

314 skip_keys, key_aliases, conditions, param_vars);
315 }
316 if (con_created && con)
317 delete con;
318}

334void Driver::_dump(realtype t, ValueSet_t* con) {
335 if (!_dumpStep) return;
336 if (t == 0) {
337 if (!fname_vars_init.empty())
338 dump(fname_vars_init, nullptr, con, true);
339 } else if (_lastStep) {
340 if (!fname_vars_last.empty())
341 dump(fname_vars_last, nullptr, con, true);
342 } else {
343 if (!fname_vars_step.empty()) {
344 std::string ifile = fname_vars_step + std::to_string(t) + ".txt";
345 }
346 }
347 if (fname_vars_step.empty())
348 _dumpStep = false;
349}

351int Driver::calculate(realtype t, N_Vector u, N_Vector u_dot, void *user_data) {
352 realtype *dxdt = N_VGetArrayPointer(u_dot);
